005package org.anarres.ipmi.protocol.codec;
006
007import java.net.DatagramPacket;
008import java.net.SocketAddress;
009import java.net.SocketException;
010import java.nio.ByteBuffer;
011import javax.annotation.Nonnull;
012import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
013import org.anarres.ipmi.protocol.packet.rmcp.Packet;
014import org.anarres.ipmi.protocol.packet.rmcp.RmcpPacket;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018/**
019 *
020 * @author shevek
021 */
022public class IpmiPacketDecoder {
023
024    private static final Logger LOG = LoggerFactory.getLogger(IpmiPacketDecoder.class);
025    private final IpmiPacketContext context;
026
027    public IpmiPacketDecoder(@Nonnull IpmiPacketContext context) {
028        this.context = context;
029    }
030
031    @Nonnull
032    public DatagramPacket encode(@Nonnull Packet packet) throws SocketException {
033        int length = packet.getWireLength(context);
034        ByteBuffer buf = ByteBuffer.allocate(length);
035        packet.toWire(context, buf);
036        return new DatagramPacket(buf.array(), buf.position(), packet.getRemoteAddress());
037    }
038
039    @Nonnull
040    public Packet decode(@Nonnull SocketAddress remoteAddress, @Nonnull ByteBuffer buf) {
041        RmcpPacket packet = new RmcpPacket();
042        packet.withRemoteAddress(remoteAddress);
043        packet.fromWire(context, buf);
044        if (buf.position() < buf.limit()) {
045            LOG.warn("Discarded " + (buf.limit() - buf.position()) + " trailing bytes in RMCP packet: " + buf);
046            buf.position(buf.limit());
047        }
048        return packet;
049    }
050}
