005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.IanaEnterpriseNumber;
012import org.anarres.ipmi.protocol.client.visitor.IpmiClientRmcpMessageHandler;
013import org.anarres.ipmi.protocol.packet.common.AbstractWireable;
014import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
015import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
016import org.anarres.ipmi.protocol.packet.rmcp.RmcpMessageClass;
017
018/**
019 * ASF RMCP Data (Enterprise number 0x4542).
020 *
021 * http://www.dmtf.org/sites/default/files/standards/documents/DSP0136.pdf
022 * http://www.dmtf.org/standards/asf
023 * [ASF2] Section 3.2.2.3 page 22.
024 * [IPMI2] Section 13.2 pages 126-127.
025 * 
026 * This class manages both the RMCP payload and the ASF payload.
027 * Implementors desiring to provide the ASF payload should override
028 * {@link #getDataWireLength()} and {@link #toWireData(ByteBuffer)}
029 * and ignore the RMCP wrapper.
030 *
031 * @author shevek
032 */
033public abstract class AbstractAsfData extends AbstractWireable implements AsfRmcpData {
034
035    public static final IanaEnterpriseNumber IANA_ENTERPRISE_NUMBER = IanaEnterpriseNumber.Alerting_Specifications_Forum;
036    // Page 33
037    private byte messageTag;    // matches request/response
038
039    @Override
040    public RmcpMessageClass getMessageClass() {
041        return RmcpMessageClass.ASF;
042    }
043
044    @Override
045    public void apply(IpmiClientRmcpMessageHandler handler, IpmiHandlerContext context) {
046        handler.handleAsfRmcpData(context, this);
047    }
048
049    public byte getMessageTag() {
050        return messageTag;
051    }
052
053    public void setMessageTag(byte messageTag) {
054        this.messageTag = messageTag;
055    }
056
057    @Override
058    public int getWireLength(IpmiPacketContext context) {
059        return 0
060                + 4 // ianaEnterpriseNumber
061                + 1 // message type
062                + 1 // message tag
063                + 1 // data length
064                + getDataWireLength();
065    }
066
067    /** Returns the length of the ASF data part of this packet. */
068    @Nonnegative
069    protected abstract int getDataWireLength();
070
071    @Override
072    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
073        buffer.putInt(IANA_ENTERPRISE_NUMBER.getNumber());
074        buffer.put(getMessageType().getCode());
075        buffer.put(getMessageTag());
076        buffer.put((byte) 0);   // reserved
077        buffer.put(UnsignedBytes.checkedCast(getDataWireLength()));
078        toWireData(buffer);
079    }
080
081    /** Serializes the ASF data into this RMCP data. */
082    protected abstract void toWireData(@Nonnull ByteBuffer buffer);
083
084    @Override
085    protected void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
086        assertWireInt(buffer, IANA_ENTERPRISE_NUMBER.getNumber(), "IANA enterprise number");
087        assertWireByte(buffer, getMessageType().getCode(), "message type code");
088        setMessageTag(buffer.get());
089        assertWireByte(buffer, (byte) 0, "reserved byte");
090        int length = buffer.get() & 0xFF;   // Can't validate until we have the data. :-(
091        fromWireData(buffer);
092        // TODO: Validate data length on read.
093    }
094
095    protected abstract void fromWireData(@Nonnull ByteBuffer buffer);
096}
