005package org.anarres.ipmi.protocol.packet.ipmi.payload;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.CheckForSigned;
009import javax.annotation.Nonnegative;
010import org.anarres.ipmi.protocol.packet.common.Code;
011
012/**
013 * [IPMI2] Section 13.27.3 page 157.
014 *
015 * @author shevek
016 */
017public enum IpmiPayloadType implements Code.Wrapper {
018
019    IPMI(0x00, 1, 0),
020    SOL(0x01, 1, 0),
021    OEM_EXPLICIT(0x02, -1, -1),
022    RMCPOpenSessionRequest(0x10, 1, 0),
023    RMCPOpenSessionResponse(0x11, 1, 0),
024    RAKPMessage1(0x12, 1, 0),
025    RAKPMessage2(0x13, 1, 0),
026    RAKPMessage3(0x14, 1, 0),
027    RAKPMessage4(0x15, 1, 0),
028    OEM0(0x20, -1, -1),
029    OEM1(0x21, -1, -1),
030    OEM2(0x22, -1, -1),
031    OEM3(0x23, -1, -1),
032    OEM4(0x24, -1, -1),
033    OEM5(0x25, -1, -1),
034    OEM6(0x26, -1, -1),
035    OEM7(0x27, -1, -1);
036    // 6 bits only.
037    private final byte code;
038    @CheckForSigned
039    private final int majorFormat;
040    @CheckForSigned
041    private final int minorFormat;
042
043    private IpmiPayloadType(@Nonnegative int code, int majorFormat, int minorFormat) {
044        this.code = UnsignedBytes.checkedCast(code);
045        this.majorFormat = majorFormat;
046        this.minorFormat = minorFormat;
047    }
048
049    @Override
050    public byte getCode() {
051        return code;
052    }
053
054    // @Nonnull public abstract IpmiPayload newIpmiPayload();
055    @Override
056    public String toString() {
057        StringBuilder buf = new StringBuilder(name());
058        if (majorFormat > 0) {
059            buf.append(" format=").append(majorFormat);
060            if (minorFormat >= 0)
061                buf.append('.').append(minorFormat);
062        }
063        return buf.toString();
064    }
065}
