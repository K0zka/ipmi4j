005package org.anarres.ipmi.protocol.packet.ipmi.payload;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnull;
010import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiPayloadHandler;
011import org.anarres.ipmi.protocol.packet.common.Bits;
012import org.anarres.ipmi.protocol.packet.common.Code;
013import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiAlgorithmUtils;
014import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiAuthenticationAlgorithm;
015import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiConfidentialityAlgorithm;
016import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiIntegrityAlgorithm;
017import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
018import org.anarres.ipmi.protocol.client.session.IpmiSession;
019import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
020
021/**
022 * [IPMI2] Section 13.17, table 13-9, page 147.
023 *
024 * @author shevek
025 */
026public class IpmiOpenSessionRequest extends AbstractTaggedIpmiPayload {
027
028    private RequestedMaximumPrivilegeLevel requestedMaximumPrivilegeLevel;
029    private int consoleSessionId;
030    private IpmiAuthenticationAlgorithm authenticationAlgorithm;
031    private IpmiIntegrityAlgorithm integrityAlgorithm;
032    private IpmiConfidentialityAlgorithm confidentialityAlgorithm;
033
034    public IpmiOpenSessionRequest() {
035    }
036
037    public IpmiOpenSessionRequest(@Nonnull IpmiSession session, @Nonnull RequestedMaximumPrivilegeLevel requestedMaximumPrivilegeLevel) {
038        this.requestedMaximumPrivilegeLevel = requestedMaximumPrivilegeLevel;
039        this.consoleSessionId = session.getConsoleSessionId();
040        this.authenticationAlgorithm = session.getAuthenticationAlgorithm();
041        this.integrityAlgorithm = session.getIntegrityAlgorithm();
042        this.confidentialityAlgorithm = session.getConfidentialityAlgorithm();
043    }
044
045    @Override
046    public IpmiPayloadType getPayloadType() {
047        return IpmiPayloadType.RMCPOpenSessionRequest;
048    }
049
050    @Override
051    public Class<? extends AbstractTaggedIpmiPayload> getRequestType() {
052        return IpmiOpenSessionRequest.class;
053    }
054
055    @Override
056    public Class<? extends AbstractTaggedIpmiPayload> getResponseType() {
057        return IpmiOpenSessionResponse.class;
058    }
059
060    @Override
061    public void apply(IpmiClientIpmiPayloadHandler handler, IpmiHandlerContext context, IpmiSession session) {
062        handler.handleOpenSessionRequest(context, session, this);
063    }
064
065    @Override
066    public int getWireLength(IpmiPacketContext context) {
067        return 32;
068    }
069
070    @Override
071    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
072        buffer.put(messageTag);
073        buffer.put(Bits.toByte(requestedMaximumPrivilegeLevel));
074        buffer.putChar((char) 0);   // reserved
075        toWireIntLE(buffer, consoleSessionId);
076        IpmiAlgorithmUtils.toWireUnchecked(buffer, authenticationAlgorithm);
077        IpmiAlgorithmUtils.toWireUnchecked(buffer, integrityAlgorithm);
078        IpmiAlgorithmUtils.toWireUnchecked(buffer, confidentialityAlgorithm);
079    }
080
081    @Override
082    protected void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
083        messageTag = buffer.get();
084        byte requestedMaximumPrivilegeLevelByte = buffer.get();
085        requestedMaximumPrivilegeLevel = Code.fromByte(RequestedMaximumPrivilegeLevel.class, (byte) (requestedMaximumPrivilegeLevelByte & RequestedMaximumPrivilegeLevel.MASK));
086        assertWireBytesZero(buffer, 2);
087        consoleSessionId = fromWireIntLE(buffer);
088        authenticationAlgorithm = IpmiAlgorithmUtils.fromWireUnchecked(buffer, IpmiAuthenticationAlgorithm.class);
089        integrityAlgorithm = IpmiAlgorithmUtils.fromWireUnchecked(buffer, IpmiIntegrityAlgorithm.class);
090        confidentialityAlgorithm = IpmiAlgorithmUtils.fromWireUnchecked(buffer, IpmiConfidentialityAlgorithm.class);
091    }
092
093    @Override
094    public void toStringBuilder(StringBuilder buf, int depth) {
095        appendHeader(buf, depth, getClass().getSimpleName());
096        depth++;
097        appendValue(buf, depth, "MessageTag", "0x" + UnsignedBytes.toString(messageTag, 16));
098        appendValue(buf, depth, "RequestedMaximumPrivilegeLevel", requestedMaximumPrivilegeLevel);
099        appendValue(buf, depth, "ConsoleSessionId", "0x" + Integer.toHexString(consoleSessionId));
100        appendValue(buf, depth, "AuthenticationAlgorithm", authenticationAlgorithm);
101        appendValue(buf, depth, "IntegrityAlgorithm", integrityAlgorithm);
102        appendValue(buf, depth, "ConfidentialityAlgorithm", confidentialityAlgorithm);
103    }
104}
