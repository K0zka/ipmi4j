005package org.anarres.ipmi.protocol.packet.ipmi.security.impl.integrity;
006
007import java.nio.ByteBuffer;
008import java.security.InvalidKeyException;
009import java.security.MessageDigest;
010import java.security.NoSuchAlgorithmException;
011import java.util.Arrays;
012import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiIntegrityAlgorithm;
013
014/**
015 * [IPMI2] Section 13.28.4, page 159.
016 *
017 * @author shevek
018 */
019public class MD5_128 implements MAC {
020
021    private final MessageDigest digest;
022    private byte[] key;
023
024    public MD5_128() throws NoSuchAlgorithmException {
025        digest = MessageDigest.getInstance("MD5");
026    }
027
028    @Override
029    public IpmiIntegrityAlgorithm getName() {
030        return IpmiIntegrityAlgorithm.MD5_128;
031    }
032
033    @Override
034    public void init(byte[] key) throws NoSuchAlgorithmException, InvalidKeyException {
035        digest.update(key);
036        this.key = key;
037    }
038
039    @Override
040    public void update(ByteBuffer input) throws IllegalStateException {
041        digest.update(input);
042    }
043
044    @Override
045    public byte[] doFinal() throws IllegalStateException {
046        byte[] out = digest.digest(key);
047        int requestedMacLength = getName().getMacLength();
048        if (requestedMacLength != out.length)
049            out = Arrays.copyOf(out, requestedMacLength);
050        return out;
051    }
052}
