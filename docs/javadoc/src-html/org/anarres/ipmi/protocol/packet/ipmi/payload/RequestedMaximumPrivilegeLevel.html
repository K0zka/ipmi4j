005package org.anarres.ipmi.protocol.packet.ipmi.payload;

007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import org.anarres.ipmi.protocol.packet.common.Bits;
010import org.anarres.ipmi.protocol.packet.common.Code;
011import org.anarres.ipmi.protocol.packet.ipmi.IpmiChannelPrivilegeLevel;

014/**
015 * [IPMI2] Section 13.20 page 150: {@link IpmiRAKPMessage1}.
016 * [IPMI2] Section 13.17 page 147: {@link IpmiOpenSessionRequest}: MAXIMUM_ALLOWED_BY_CIPHER.
017 * [IPMI2] Section 22.18, table 22-23, page 297: {@link SetSessionPrivilegeLevelRequest}: UNCHANGED.
018 *
019 * @see IpmiChannelPrivilegeLevel
020 */
021// TODO: -> IpmiPrivilegeLevel
022public enum RequestedMaximumPrivilegeLevel implements Bits.Wrapper, Code.Wrapper {
023
024    UNSPECIFIED(0),
025    CALLBACK(1),
026    USER(2),
027    OPERATOR(3),
028    ADMINISTRATOR(4),
029    OEM(5);
030    public static final int MASK = 0x0F;
031    private final Bits bits;
032
033    private RequestedMaximumPrivilegeLevel(@Nonnegative int code) {
034        // Do NOT use the MASK constant here - it's initialized AFTER the enum constants.
035        this.bits = new Bits(0, 0xF, UnsignedBytes.checkedCast(code));
036    }
037
038    @Override
039    public Bits getBits() {
040        return bits;
041    }
042
043    @Override
044    public byte getCode() {
045        return UnsignedBytes.checkedCast(getBits().getByteValue());
046    }
047}
