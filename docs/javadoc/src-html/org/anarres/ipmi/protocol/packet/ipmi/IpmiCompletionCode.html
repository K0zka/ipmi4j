005package org.anarres.ipmi.protocol.packet.ipmi;
006
007import com.google.common.primitives.UnsignedBytes;
008import org.anarres.ipmi.protocol.packet.common.Code;
009
010/**
011 * [IPMI2] Section 5.2 page 43.
012 *
013 * @author shevek
014 */
015public enum IpmiCompletionCode implements Code.DescriptiveWrapper {
016
017    Normal(0x00, "Command Completed Normally."),
018    NodeBusy(0xC0, "Node Busy. Command could not be processed because command processing resources are temporarily unavailable."),
019    InvalidCommand(0xC1, "Invalid Command. Used to indicate an unrecognized or unsupported command."),
020    InvalidCommandForLUN(0xC2, "Command invalid for given LUN."),
021    Timeout(0xC3, "Timeout while processing command. Response unavailable."),
022    OutOfSpace(0xC4, "Out of space. Command could not be completed because of a lack of storage space required to execute the given command operation."),
023    ReservationCanceled(0xC5, "Reservation Canceled or Invalid Reservation ID."),
024    RequestDataTruncated(0xC6, "Request data truncated."),
025    RequestDataLengthInvalid(0xC7, "Request data length invalid."),
026    RequestDataLengthFieldLimitExceeded(0xC8, "Request data field length limit exceeded."),
027    ParameterOutOfRange(0xC9, "Parameter out of range. One or more parameters in the data field of the Request are out of range."),
028    CannotReturnRequestedNumberOfBytes(0xCA, "Cannot return number of requested data bytes."),
029    NotPresent(0xCB, "Requested Sensor, data, or record not present."),
030    InvalidDataField(0xCC, "Invalid data field in Request."),
031    IllegalCommandForTarget(0xCD, "Command illegal for specified sensor or record type."),
032    CommandResponseNotAvailable(0xCE, "Command response could not be provided."),
033    CannotExecuteDuplicatedRequest(0xCF, "Cannot execute duplicated request."),
034    CommandResponseNotAvailable_SDRUpdate(0xD0, "Command response could not be provided. SDR Repository in update mode."),
035    CommandResponseNotAvailable_FirmwareUpdate(0xD1, "Command response could not be provided. Device in firmware update mode."),
036    CommandResponseNotAvailable_BMCInitialization(0xD2, "Command response could not be provided. BMC initialization or initialization agent in progress."),
037    DestinationUnavailable(0xD3, "Destination unavailable. Cannot deliver request to selected destination."),
038    InsufficientPrivilege(0xD4, "Cannot execute command due to insufficient privilege level or other security-based restriction."),
039    IllegalState(0xD5, "Cannot execute command. Command, or request parameter(s), not supported in present state."),
040    IllegalParameterForSubFunctionState(0xD6, "Cannot execute command. Parameter is illegal because command sub-function has been disabled or is unavailable."),
041    UnspecifiedError(0xFF, "Unspecified error.");
042    /*
043     * DEVICE-SPECIFIC (OEM) CODES 01h-7Eh 
044     * 01h-7Eh Device specific (OEM) completion codes. This range is used for command-
045     * specific codes that are also specific for a particular device and version. A-priori
046     * knowledge of the device command set is required for interpretation of these
047     * codes.
048     * COMMAND-SPECIFIC CODES 80h-BEh 
049     * 80h-BEh Standard command-specific codes. This range is reserved for command-
050     * specific completion codes for commands specified in this document.
051     */
052    private final byte code;
053    private final String description;
054
055    private IpmiCompletionCode(int code, String description) {
056        this.code = UnsignedBytes.checkedCast(code);
057        this.description = description;
058    }
059
060    @Override
061    public byte getCode() {
062        return code;
063    }
064
065    @Override
066    public String getDescription() {
067        return description;
068    }
069
070    @Override
071    public String toString() {
072        return Code.Utils.toString(this);
073    }
074}
