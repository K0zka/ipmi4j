005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientAsfMessageHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.common.Bits;
014import org.anarres.ipmi.protocol.packet.common.Code;
015
016/**
017 * [ASF2] Section 3.2.4.5, page 39.
018 *
019 * @author shevek
020 */
021public class AsfSystemStateResponseData extends AbstractAsfData {
022
023    // TODO: Don't implement Bits.Wrapper
024    /** [ASF2] Section 3.2.4.5, page 39. */
025    public enum SystemState implements Code.Wrapper {
026
027        STATE_WORKING(0b0000),
028        STATE_S1(0b0001),
029        STATE_S2(0b0010),
030        STATE_S3(0b0011),
031        STATE_S4(0b0100),
032        STATE_S5(0b0101),
033        STATE_S4_S5(0b0110),
034        STATE_G3(0b0111),
035        STATE_G1(0b1000),
036        STATE_S5_OVERRIDE(0b1001),
037        STATE_LEGACY_ON(0b1011),
038        STATE_LEGACY_OFF(0b1100),
039        STATE_UNKNOWN(0b1110),
040        STATE_RESERVED(0b1110);
041        private final byte code;
042
043        private SystemState(@Nonnegative int code) {
044            this.code = UnsignedBytes.checkedCast(code);
045        }
046
047        @Override
048        public byte getCode() {
049            return code;
050        }
051    }
052
053    // TODO: Don't implement Bits.Wrapper
054    public static enum WatchdogState implements Bits.Wrapper, Code.Wrapper {
055
056        WATCHDOG_TIMER_NOT_EXPIRED(Bits.forBitIndex(0, 0, false)),
057        WATCHDOG_TIMER_EXPIRED(Bits.forBitIndex(0, 0, true));
058        private final Bits bits;
059
060        private WatchdogState(@Nonnull Bits bits) {
061            this.bits = bits;
062        }
063
064        @Override
065        public Bits getBits() {
066            return bits;
067        }
068
069        @Override
070        public byte getCode() {
071            return (byte) getBits().getByteValue();
072        }
073    }
074    private SystemState systemState = SystemState.STATE_UNKNOWN;
075    private WatchdogState watchdogState = WatchdogState.WATCHDOG_TIMER_NOT_EXPIRED;
076
077    @Nonnull
078    public SystemState getSystemState() {
079        return systemState;
080    }
081
082    @Nonnull
083    public AsfSystemStateResponseData withSystemState(SystemState systemState) {
084        this.systemState = systemState;
085        return this;
086    }
087
088    @Nonnull
089    public WatchdogState getWatchdogState() {
090        return watchdogState;
091    }
092
093    @Nonnull
094    public AsfSystemStateResponseData withWatchdogState(WatchdogState watchdogState) {
095        this.watchdogState = watchdogState;
096        return this;
097    }
098
099    @Override
100    public AsfRmcpMessageType getMessageType() {
101        return AsfRmcpMessageType.SystemStateResponse;
102    }
103
104    @Override
105    public void apply(IpmiClientAsfMessageHandler handler, IpmiHandlerContext context) {
106        handler.handleAsfSystemStateResponseData(context, this);
107    }
108
109    @Override
110    protected int getDataWireLength() {
111        return 4;
112    }
113
114    @Override
115    protected void toWireData(ByteBuffer buffer) {
116        buffer.put(getSystemState().getCode());
117        buffer.put(getWatchdogState().getCode());
118        buffer.put(new byte[2]);    // Reserved
119    }
120
121    @Override
122    protected void fromWireData(ByteBuffer buffer) {
123        withSystemState(Code.fromBuffer(SystemState.class, buffer));
124        withWatchdogState(Code.fromBuffer(WatchdogState.class, buffer));
125        assertWireBytesZero(buffer, 2);
126    }
127}
