042    /**
043     * @see AbstractIpmiCommand#fromWireUnchecked(IpmiPacketContext, ByteBuffer)
044     */
045    @Nonnull
046    protected static IpmiPayload newPayload(@Nonnull ByteBuffer buffer, @Nonnull IpmiPayloadType payloadType) {
047        switch (payloadType) {
048            case RMCPOpenSessionRequest:
049                return new IpmiOpenSessionRequest();
050            case RMCPOpenSessionResponse:
051                return new IpmiOpenSessionResponse();
052            case RAKPMessage1:
053                return new IpmiRAKPMessage1();
054            case RAKPMessage2:
055                return new IpmiRAKPMessage2();
056            case RAKPMessage3:
057                return new IpmiRAKPMessage3();
058            case RAKPMessage4:
059                return new IpmiRAKPMessage4();
060            case IPMI:
061                int position = buffer.position();
062                int networkFunctionByte = UnsignedBytes.toInt(buffer.get(position + 1));
063                // LOG.info("NetworkFunctionByte is 0x" + Integer.toHexString(networkFunctionByte));
064                IpmiNetworkFunction networkFunction = Code.fromInt(IpmiNetworkFunction.class, (networkFunctionByte >>> 2) & ~1);
065                byte commandNameByte = buffer.get(position + 5);
066                IpmiCommandName commandName = IpmiCommandName.fromByte(networkFunction, commandNameByte);
067                boolean isResponse = ((networkFunctionByte >>> 2) & 1) != 0;
068                return isResponse ? commandName.newResponseMessage() : commandName.newRequestMessage();
069            case SOL:
070                return new SOLMessage();
071            case OEM_EXPLICIT:
072                return new OemExplicit();
073            default:
074                throw new UnsupportedOperationException("Unsupported payload type " + payloadType);
075        }
076    }

146    @Override
147    public <T> T getEncapsulated(Class<T> type) {
148        if (type.isInstance(this))

151    }

153    @Override
154    public void toStringBuilder(StringBuilder buf, int depth) {
155        appendHeader(buf, depth, "IpmiSessionData");
156        depth++;
157        appendValue(buf, depth, "SessionId", "0x" + Integer.toHexString(getIpmiSessionId()));
158        appendValue(buf, depth, "SessionSequenceNumber", "0x" + Integer.toHexString(getIpmiSessionSequenceNumber()));
159        appendChild(buf, depth, "IpmiPayload", getIpmiPayload());
160    }
