005package org.anarres.ipmi.protocol.packet.ipmi;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import org.anarres.ipmi.protocol.packet.common.Code;
010import org.anarres.ipmi.protocol.packet.ipmi.command.messaging.GetChannelAuthenticationCapabilitiesResponse;
011
012/**
013 * [IPMI2] Section 13.6 page 133, right margin of table.
014 * NOT [IPMI2] Section 13.28 page 157.
015 * Codes used as bit offsets in GetChannelAuthenticationCapabilitiesResponse.
016 *
017 * @see GetChannelAuthenticationCapabilitiesResponse
018 * @author shevek
019 */
020public enum IpmiSessionAuthenticationType implements Code.Wrapper {
021
022    /** This means IPMI v1.5 */
023    NONE(0),
024    /** This means IPMI v1.5 */
025    MD2(1),
026    /** This means IPMI v1.5 */
027    MD5(2),
028    /** This means IPMI v1.5 */
029    PASSWORD(4),
030    /** This means IPMI v1.5 */
031    OEM_PROPRIETARY(5),
032    /** This means IPMI v2.0 */
033    RMCPP(6);
034    private final byte code;
035
036    private IpmiSessionAuthenticationType(@Nonnegative int code) {
037        this.code = UnsignedBytes.checkedCast(code);
038    }
039
040    @Override
041    public byte getCode() {
042        return code;
043    }
044}
