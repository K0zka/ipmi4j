005package org.anarres.ipmi.protocol.packet.ipmi.command.global;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import java.util.EnumSet;
010import java.util.Set;
011import javax.annotation.Nonnegative;
012import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
013import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
014import org.anarres.ipmi.protocol.packet.common.Bits;
015import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
016import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiResponse;
017
018/**
019 * [IPMI2] Section 20.1, table 20-2, page 244.
020 *
021 * @author shevek
022 */
023public class GetDeviceIdResponse extends AbstractIpmiResponse {
024
025    public static enum DeviceSupport implements Bits.Wrapper {
026
027        Chassis(7),
028        Bridge(6),
029        IPMBEventGenerator(5),
030        IPMBEventReceiver(4),
031        FRUInventoryDevice(3),
032        SELDevice(2),
033        SDRRepositoryDevice(1),
034        SensorDevice(0);
035        private final Bits bits;
036
037        private DeviceSupport(@Nonnegative int index) {
038            this.bits = Bits.forBitIndex(0, index);
039        }
040
041        @Override
042        public Bits getBits() {
043            return bits;
044        }
045    }
046    public byte deviceId;
047    public boolean deviceProvidesDeviceSDRs;
048    public int deviceRevision;
049    public boolean deviceAvailable;
050    public int deviceFirmwareRevisionMajor;
051    public int deviceFirmwareRevisionMinor;
052    public int deviceIpmiRevision;
053    public Set<DeviceSupport> deviceSupport = EnumSet.noneOf(DeviceSupport.class);
054    public int oemEnterpriseNumber;
055    public char productId;
056    public Integer auxiliaryFirmwareRevisionInformation;
057
058    @Override
059    public IpmiCommandName getCommandName() {
060        return IpmiCommandName.GetDeviceID;
061    }
062
063    @Override
064    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
065        handler.handleGetDeviceIdResponse(context, this);
066    }
067
068    @Override
069    protected int getResponseDataWireLength() {
070        return auxiliaryFirmwareRevisionInformation == null ? 12 : 16;
071    }
072
073    @Override
074    protected void toWireData(ByteBuffer buffer) {
075        if (toWireCompletionCode(buffer))   // 1
076            return;
077        buffer.put(deviceId);   // 2
078        buffer.put(setBit((byte) (deviceRevision & 0xF), 7, deviceProvidesDeviceSDRs)); // 3
079        buffer.put(setBit((byte) (deviceFirmwareRevisionMajor & 0x7F), 7, deviceAvailable));    // 4
080        buffer.put(toWireBcdLE(UnsignedBytes.checkedCast(deviceFirmwareRevisionMinor)));   // 5
081        buffer.put(toWireBcdLE(UnsignedBytes.checkedCast(deviceIpmiRevision)));    // 6
082        buffer.put(Bits.toByte(deviceSupport)); // 7
083        toWireOemIanaLE3(buffer, oemEnterpriseNumber); // 8:10
084        toWireCharLE(buffer, productId);
085        if (auxiliaryFirmwareRevisionInformation != null)
086            buffer.putInt(auxiliaryFirmwareRevisionInformation);
087    }
088
089    @Override
090    protected void fromWireData(ByteBuffer buffer) {
091        if (fromWireCompletionCode(buffer)) // 1
092            return;
093        deviceId = buffer.get();    // 2
094        byte b = buffer.get();  // 3
095        deviceProvidesDeviceSDRs = getBit(b, 7);
096        deviceRevision = b & 0xF;
097        b = buffer.get();   // 4
098        deviceAvailable = getBit(b, 7);
099        deviceFirmwareRevisionMajor = b & 0x7F;
100        deviceFirmwareRevisionMinor = fromWireBcdLE(buffer.get());  // 5
101        deviceIpmiRevision = fromWireBcdLE(buffer.get());   // 6
102        deviceSupport = Bits.fromBuffer(DeviceSupport.class, buffer, 1);    // 7
103        oemEnterpriseNumber = fromWireOemIanaLE3(buffer);  // 8:10
104        productId = fromWireCharLE(buffer); // 11:12
105        if (buffer.remaining() >= 4)
106            auxiliaryFirmwareRevisionInformation = buffer.getInt();
107        else
108            auxiliaryFirmwareRevisionInformation = null;
109    }
110
111    @Override
112    public void toStringBuilder(StringBuilder buf, int depth) {
113        super.toStringBuilder(buf, depth);
114        appendValue(buf, depth, "DeviceId", "0x" + UnsignedBytes.toString(deviceId, 16));
115        appendValue(buf, depth, "DeviceAvailable", deviceAvailable);
116        appendValue(buf, depth, "ManufacturerIanaEnterpriseNumber", toStringOemIana(oemEnterpriseNumber));
117        appendValue(buf, depth, "ProductId", productId);
118        appendValue(buf, depth, "DeviceRevision", "0x" + Integer.toHexString(deviceRevision));
119        appendValue(buf, depth, "DeviceFirmwareRevision", deviceFirmwareRevisionMajor + "." + deviceFirmwareRevisionMinor);
120        appendValue(buf, depth, "AuxiliaryFirmwareRevisionInformation", auxiliaryFirmwareRevisionInformation);
121        appendValue(buf, depth, "DeviceIpmiRevision", deviceIpmiRevision);  // divide by 10? 20 = 2.0, 15 = 1.5
122        appendValue(buf, depth, "DeviceSupport", deviceSupport);
123        appendValue(buf, depth, "DeviceProvidesDeviceSDRs", deviceProvidesDeviceSDRs);
124    }
125}
