025/**
026 *
027 * @author shevek
028 */
029public abstract class AbstractPacket extends AbstractWireable implements Packet {
030
031    private static final Logger LOG = LoggerFactory.getLogger(AbstractPacket.class);
032    private SocketAddress remoteAddress;
033    protected static final int RMCP_HEADER_LENGTH = 4;
034    private final RmcpVersion version = RmcpVersion.ASF_RMCP_2_0;
035    /** 0xFF means do not generate an ACK. */
036    private byte sequenceNumber = (byte) 0xFF;
037    private RmcpMessageClass messageClass;
038    private RmcpMessageRole messageRole = RmcpMessageRole.REQ;
039    /** Data for a message, null for an ack. */
040    private RmcpData data;
041
042    @Override
043    public SocketAddress getRemoteAddress() {
044        return remoteAddress;
045    }
046
047    @Nonnull
048    public AbstractPacket withRemoteAddress(@Nonnull SocketAddress remoteAddress) {
049        this.remoteAddress = remoteAddress;
050        return this;
051    }
052
053    @Nonnull
054    public RmcpVersion getVersion() {
055        return version;
056    }
057
058    public byte getSequenceNumber() {
059        return sequenceNumber;
060    }
061
062    @Nonnull
063    public AbstractPacket withSequenceNumber(byte sequenceNumber) {
064        this.sequenceNumber = sequenceNumber;
065        return this;
066    }
067
068    @Nonnull
069    public RmcpMessageClass getMessageClass() {
070        return messageClass;
071    }
072
073    @Nonnull
074    public AbstractPacket withMessageClass(RmcpMessageClass messageClass) {
075        this.messageClass = messageClass;
076        return this;
077    }
078
079    @Nonnull
080    public RmcpMessageRole getMessageRole() {
081        return messageRole;
082    }
083
084    @Nonnull
085    public AbstractPacket withMessageRole(RmcpMessageRole messageRole) {
086        this.messageRole = messageRole;
087        return this;
088    }
089
090    @Override
091    public RmcpData getData() {
092        return data;
093    }
094
095    @Override
096    public <T extends RmcpData> T getData(Class<T> type) {
097        return type.cast(getData());
098    }
099
100    @Override
101    public Packet withData(@Nonnull RmcpData data) {
102        withMessageClass(data.getMessageClass());
103        this.data = data;
104        return this;
105    }
106
107    @Override
108    public <T> T getEncapsulated(Class<T> type) {
109        if (type.isInstance(this))
110            return type.cast(this);
111        return Encapsulation.Utils.getEncapsulated(type, getData());
112    }
113
114    @Override
115    public void apply(IpmiClientRmcpMessageHandler handler, IpmiHandlerContext context) {
116        getData().apply(handler, context);
117    }
118
119    @Nonnegative
120    protected int getRawWireLength(@Nonnull IpmiPacketContext context) {
121        return RMCP_HEADER_LENGTH + getData().getWireLength(context);
122    }
123
124    /**
125     * RMCP Packet Header.
126     * [ASF2] Section 3.2.2.2 page 21.
127     */
128    private void toWireHeader(@Nonnull ByteBuffer buffer) {
129        // DSP0136 page 22
130        buffer.put(getVersion().getCode());
131        buffer.put((byte) 0x00); // ASF reserved
132        buffer.put(getSequenceNumber());
133        byte messageClassByte = getMessageClass().getCode();
134        if (RmcpMessageRole.ACK.equals(getMessageRole()))
135            messageClassByte |= 0x80;
136        buffer.put(messageClassByte);
137    }
138
139    protected void toWireRaw(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer) {
140        toWireHeader(buffer);
141        getData().toWire(context, buffer);
142    }
143
144    @Nonnull
145    private RmcpMessageClass fromWireHeader(@Nonnull ByteBuffer buffer) {
146        assertWireByte(buffer, getVersion().getCode(), "RMCP version");
147        assertWireByte(buffer, (byte) 0, "reserved field");
148        withSequenceNumber(buffer.get());
149        byte messageClassByte = buffer.get();
150        withMessageClass(Code.fromInt(RmcpMessageClass.class, messageClassByte & 0x7f));
151        withMessageRole(Code.fromInt(RmcpMessageRole.class, (messageClassByte >>> 7) & 0x1));
152        return getMessageClass();
153    }
154
155    protected void fromWireRaw(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer) {
156        RmcpMessageClass messageClass = fromWireHeader(buffer);
157        RmcpData data;
158
159        int position = buffer.position();
160        switch (messageClass) {
161            case ASF:
162                int enterpriseNumber = buffer.getInt(buffer.position());
163                if (enterpriseNumber != AbstractAsfData.IANA_ENTERPRISE_NUMBER.getNumber())
164                    throw new IllegalArgumentException("Unknown enterprise number 0x" + Integer.toHexString(enterpriseNumber));
165                AsfRmcpMessageType messageType = Code.fromByte(AsfRmcpMessageType.class, buffer.get(buffer.position() + 4));
166                data = messageType.newPacketData();
167                break;
168            case IPMI:
169                byte authenticationTypeByte = buffer.get(buffer.position());
170                IpmiSessionAuthenticationType authenticationType = Code.fromByte(IpmiSessionAuthenticationType.class, authenticationTypeByte);
171                if (authenticationType == IpmiSessionAuthenticationType.RMCPP)
172                    data = new Ipmi20SessionWrapper();
173                else
174                    data = new Ipmi15SessionWrapper();
175                break;
176            case OEM:
177                data = new OEMRmcpMessage();
178                break;
179            default:
180                throw new IllegalArgumentException("Can't decode buffer: Unknown MessageClass " + messageClass);
181        }
182        buffer.position(position);
183
184        // LOG.info("Buffer position pre-data is " + buffer.position());
185        withData(data);
186        data.fromWire(context, buffer);
187    }
188
189    @Override
190    public void toStringBuilder(StringBuilder buf, int depth) {
191        appendValue(buf, depth, "RmcpSequenceNumber", UnsignedBytes.toInt(getSequenceNumber()));
192        appendValue(buf, depth, "RmcpMessageClass", getMessageClass());
193        appendValue(buf, depth, "RmcpMessageRole", getMessageRole());
194    }
195}
