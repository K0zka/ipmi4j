001package org.anarres.ipmi.protocol.packet.rmcp;
002
003import java.net.SocketAddress;
004import javax.annotation.Nonnull;
005import org.anarres.ipmi.protocol.client.visitor.IpmiClientRmcpMessageHandler;
006import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
007import org.anarres.ipmi.protocol.packet.common.Wireable;
008
009/**
010 *
011 * @author shevek
012 */
013public interface Packet extends Wireable, Encapsulation {
014
015    /**
016     * Returns the remote address from which this packet was received, or to which it will be sent.
017     */
018    @Nonnull
019    public SocketAddress getRemoteAddress();
020
021    /**
022     * Returns the RMCP payload of this packet.
023     */
024    // @CheckForNull
025    public RmcpData getData();
026
027    public <T extends RmcpData> T getData(@Nonnull Class<T> type);
028
029    /**
030     * Sets the RMCP payload of this packet.
031     */
032    @Nonnull
033    public Packet withData(@Nonnull RmcpData data);
034
035    public void apply(@Nonnull IpmiClientRmcpMessageHandler handler, @Nonnull IpmiHandlerContext context);
036}
