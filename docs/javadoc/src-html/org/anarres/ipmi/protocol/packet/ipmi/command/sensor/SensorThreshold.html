005package org.anarres.ipmi.protocol.packet.ipmi.command.sensor;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.CheckForNull;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.packet.common.Code;
012import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiCommand;
013
014/**
015 * A sensor threshold, encapsulating a {@link SensorBoundary} and a {@link SensorCriticality}.
016 *
017 * @author shevek
018 */
019public enum SensorThreshold implements Code.Wrapper {
020
021    LowerNonCritical(SensorBoundary.Lower, SensorCriticality.NonCritical, 0),
022    LowerCritical(SensorBoundary.Lower, SensorCriticality.Critical, 1),
023    LowerNonRecoverable(SensorBoundary.Lower, SensorCriticality.NonRecoverable, 2),
024    UpperNonCritical(SensorBoundary.Upper, SensorCriticality.NonCritical, 3),
025    UpperCritical(SensorBoundary.Upper, SensorCriticality.Critical, 4),
026    UpperNonRecoverable(SensorBoundary.Upper, SensorCriticality.NonRecoverable, 5);
027    private final SensorBoundary boundary;
028    private final SensorCriticality criticality;
029    private final byte code;
030
031    private SensorThreshold(@Nonnull SensorBoundary boundary, @Nonnull SensorCriticality criticality, int code) {
032        this.boundary = boundary;
033        this.criticality = criticality;
034        this.code = UnsignedBytes.checkedCast(code);
035        if (code != ordinal())
036            throw new IllegalArgumentException("Bad ordinal on " + this);
037    }
038
039    @Nonnull
040    public SensorBoundary getBoundary() {
041        return boundary;
042    }
043
044    @Nonnull
045    public SensorCriticality getCriticality() {
046        return criticality;
047    }
048
049    @Override
050    public byte getCode() {
051        return code;
052    }
053
054    /* pp */ static void toBuffer(@Nonnull ByteBuffer buffer, @CheckForNull Byte value) {
055        if (value == null)
056            buffer.put((byte) 0);
057        else
058            buffer.put(value.byteValue());
059    }
060
061    @CheckForNull
062    /* pp */ static Byte fromBuffer(@Nonnull ByteBuffer buffer, @Nonnull byte flags, int bit) {
063        byte value = buffer.get();
064        if (AbstractIpmiCommand.getBit(flags, bit))
065            return value;
066        else
067            return null;
068    }
069
070    @Nonnull
071    /* pp */ static String toString(@CheckForNull Byte value) {
072        if (value == null)
073            return "Not specified";
074        return UnsignedBytes.toString(value);
075    }
076}
