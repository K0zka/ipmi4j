005package org.anarres.ipmi.protocol.packet.ipmi;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import javax.annotation.Nonnull;
010import org.anarres.ipmi.protocol.packet.asf.AsfRsspSessionStatus;
011import org.anarres.ipmi.protocol.packet.common.Code;
012import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayloadType;
013import static org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayloadType.*;
014
015/**
016 * [IPMI2] Section 13.24, table 13-15, page 154.
017 *
018 * @see AsfRsspSessionStatus
019 * @author shevek
020 */
021public enum IpmiSessionStatus implements Code.Wrapper {
022
023    NO_ERROR(0x00, RMCPOpenSessionRequest, RAKPMessage2, RAKPMessage3, RAKPMessage4),
024    INSUFFICIENT_RESOURCES(0x01, RMCPOpenSessionRequest, RAKPMessage2, RAKPMessage3, RAKPMessage4),
025    INVALID_SESSION_ID(0x02, RMCPOpenSessionRequest, RAKPMessage2, RAKPMessage3, RAKPMessage4),
026    INVALID_PAYLOAD_TYPE(0x03, RMCPOpenSessionRequest),
027    INVALID_AUTHENTICATION_ALGORITHM(0x04, RMCPOpenSessionRequest),
028    INVALID_INTEGRITY_ALGORITHM(0x05, RMCPOpenSessionRequest),
029    NO_MATCHING_AUTHENTICATION_PAYLOAD(0x06, RMCPOpenSessionRequest),
030    NO_MATCHING_INTEGRITY_PAYLOAD(0x07, RMCPOpenSessionRequest),
031    INACTIVE_SESSION_ID(0x08, RAKPMessage2, RAKPMessage3, RAKPMessage4),
032    INVALID_ROLE(0x09, RMCPOpenSessionRequest, RAKPMessage2),
033    UNAUTHORIZED_ROLE(0x0A, RAKPMessage2),
034    INSUFFICIENT_RESOURCES_FOR_ROLE(0x0B, RAKPMessage2),
035    INVALID_NAME_LENGTH(0x0C, RAKPMessage2),
036    UNAUTHORIZED_NAME(0x0D, RAKPMessage2),
037    UNAUTHORIZED_GUID(0x0E, RAKPMessage3),
038    INVALID_INTEGRITY_CHECK_VALUE(0x0F, RAKPMessage3, RAKPMessage4),
039    INVALID_CONFIDENTIALITY_ALGORITHM(0x10, RMCPOpenSessionRequest),
040    NO_CIPHER_SUITE_MATCH(0x11, RMCPOpenSessionRequest),
041    ILLEGAL_OR_UNRECOGNIZED_PARAMETER(0x12, RMCPOpenSessionRequest, RAKPMessage2, RAKPMessage3, RAKPMessage4);
042    private final byte code;
043    private final IpmiPayloadType[] allowedPayloadTypes;
044
045    private IpmiSessionStatus(@Nonnegative int code, @Nonnull IpmiPayloadType... allowedPayloadTypes) {
046        this.code = UnsignedBytes.checkedCast(code);
047        this.allowedPayloadTypes = allowedPayloadTypes;
048    }
049
050    @Override
051    public byte getCode() {
052        return code;
053    }
054}
