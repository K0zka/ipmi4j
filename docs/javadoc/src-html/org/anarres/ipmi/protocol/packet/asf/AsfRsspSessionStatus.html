005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import javax.annotation.Nonnull;
010import org.anarres.ipmi.protocol.packet.common.Code;
011import org.anarres.ipmi.protocol.packet.ipmi.IpmiSessionStatus;
012import static org.anarres.ipmi.protocol.packet.asf.AsfRmcpMessageType.*;
013
014/**
015 * RSSP and RAKP Status Codes.
016 * 
017 * [ASF] Section 3.2.3.5.1 page 32.
018 *
019 * @see IpmiSessionStatus
020 * @author shevek
021 */
022public enum AsfRsspSessionStatus implements Code.Wrapper {
023
024    NO_ERROR(0x00, OpenSessionResponse, CloseSessionResponse, RAKPMessage2, RAKPMessage3),
025    INSUFFICIENT_RESOURCES(0x01, OpenSessionResponse),
026    INVALID_SESSION_ID(0x02, OpenSessionResponse, CloseSessionResponse, RAKPMessage2, RAKPMessage3),
027    INVALID_PAYLOAD_TYPE(0x03, OpenSessionResponse),
028    INVALID_AUTHENTICATION_ALGORITHM(0x04, OpenSessionResponse),
029    INVALID_INTEGRITY_ALGORITHM(0x05, OpenSessionResponse),
030    NO_MATCHING_AUTHENTICATION_PAYLOAD(0x06, OpenSessionResponse),
031    NO_MATCHING_INTEGRITY_PAYLOAD(0x07, OpenSessionResponse),
032    INACTIVE_SESSION_ID(0x08, CloseSessionResponse, RAKPMessage2, RAKPMessage3),
033    INVALID_ROLE(0x09, RAKPMessage2),
034    UNAUTHORIZED_ROLE(0x0A, RAKPMessage2),
035    INSUFFICIENT_RESOURCES_FOR_ROLE(0x0B, RAKPMessage2),
036    INVALID_NAME_LENGTH(0x0C, RAKPMessage2),
037    UNAUTHORIZED_NAME(0x0D, RAKPMessage2),
038    UNAUTHORIZED_GUID(0x0E, RAKPMessage3),
039    INVALID_INTEGRITY_CHECK_VALUE(0x0F, RAKPMessage3);
040    private final byte code;
041    private final AsfRmcpMessageType[] messageTypes;
042
043    private AsfRsspSessionStatus(@Nonnegative int code, @Nonnull AsfRmcpMessageType... messageTypes) {
044        this.code = UnsignedBytes.checkedCast(code);
045        this.messageTypes = messageTypes;
046    }
047
048    @Override
049    public byte getCode() {
050        return code;
051    }
052}
