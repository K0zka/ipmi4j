005package org.anarres.ipmi.protocol.packet.ipmi.command.sdr;
006
007import java.nio.ByteBuffer;
008import java.util.EnumSet;
009import java.util.Set;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.common.Bits;
014import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
015import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiResponse;
016import org.anarres.ipmi.protocol.packet.ipmi.command.sel.GetSELInfoResponse;
017
018/**
019 * [IPMI2] Section 33.9, table 33-3, page 441.
020 *
021 * @see GetSELInfoResponse
022 * @author shevek
023 */
024public class GetSDRRepositoryInfoResponse extends AbstractIpmiResponse {
025
026    public static enum SupportedOperation implements Bits.Wrapper {
027
028        OverflowFlag(Bits.forBitIndex(0, 7)),
029        // TODO -> two bits?
030        RepositoryUpdateModalityUnspecified(new Bits(0, 0b11 << 5, 0b00 << 5)),
031        RepositoryUpdateModalityNonModal(new Bits(0, 0b11 << 5, 0b01 << 5)),
032        RepositoryUpdateModalityModal(new Bits(0, 0b11 << 5, 0b10 << 5)),
033        RepositoryUpdateModalityBoth(new Bits(0, 0b11 << 5, 0b11 << 5)),
034        DeleteSDRCommandSupported(Bits.forBitIndex(0, 3)),
035        PartialAddSDRCommandSupported(Bits.forBitIndex(0, 2)),
036        ReserveSDRRepositoryCommandSupported(Bits.forBitIndex(0, 1)),
037        GetSDRRepositoryAllocationInformationCommandSupported(Bits.forBitIndex(0, 0));
038        private final Bits bits;
039
040        /* pp */ SupportedOperation(@Nonnull Bits bits) {
041            this.bits = bits;
042        }
043
044        @Override
045        public Bits getBits() {
046            return bits;
047        }
048    }
049    public char recordCount;
050    public char freeSpace;
051    public int mostRecentAdditionTimestamp;
052    public int mostRecentDeletionTimestamp;
053    public Set<SupportedOperation> supportedOperations = EnumSet.noneOf(SupportedOperation.class);
054
055    @Override
056    public IpmiCommandName getCommandName() {
057        return IpmiCommandName.GetSDRRepositoryInfo;
058    }
059
060    @Override
061    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
062        handler.handleGetSDRRepositoryInfoResponse(context, this);
063    }
064
065    @Override
066    protected int getResponseDataWireLength() {
067        return 15;
068    }
069
070    @Override
071    protected void toWireData(ByteBuffer buffer) {
072        if (toWireCompletionCode(buffer))
073            return;
074        buffer.put((byte) 0x51);
075        toWireCharLE(buffer, recordCount);
076        toWireCharLE(buffer, freeSpace);
077        toWireIntLE(buffer, mostRecentAdditionTimestamp);
078        toWireIntLE(buffer, mostRecentDeletionTimestamp);
079        buffer.put(Bits.toByte(supportedOperations));
080    }
081
082    @Override
083    protected void fromWireData(ByteBuffer buffer) {
084        if (fromWireCompletionCode(buffer))
085            return;
086        assertWireByte(buffer, (byte) 0x51, "SDR Version");
087        recordCount = fromWireCharLE(buffer);
088        freeSpace = fromWireCharLE(buffer);
089        mostRecentAdditionTimestamp = fromWireIntLE(buffer);
090        mostRecentDeletionTimestamp = fromWireIntLE(buffer);
091        supportedOperations = Bits.fromBuffer(SupportedOperation.class, buffer, 1);
092    }
093
094    @Override
095    public void toStringBuilder(StringBuilder buf, int depth) {
096        super.toStringBuilder(buf, depth);
097        appendValue(buf, depth, "RecordCount", (int) recordCount);
098        appendValue(buf, depth, "FreeSpace", "0x" + Integer.toHexString(freeSpace));
099        appendValue(buf, depth, "MostRecentAdditionTimestamp", "0x" + Integer.toHexString(mostRecentAdditionTimestamp));
100        appendValue(buf, depth, "MostRecentDeletionTimestamp", "0x" + Integer.toHexString(mostRecentDeletionTimestamp));
101        appendValue(buf, depth, "SupportedOperations", supportedOperations);
102    }
103}
