005package org.anarres.ipmi.protocol.packet.ipmi.payload;

007import com.google.common.base.Charsets;
008import com.google.common.primitives.UnsignedBytes;
009import java.nio.ByteBuffer;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiPayloadHandler;
012import org.anarres.ipmi.protocol.packet.common.Bits;
013import org.anarres.ipmi.protocol.packet.common.Code;

019/**
020 * [IPMI2] Section 13.20 page 150.
021 *
022 * @author shevek
023 */
024public class IpmiRAKPMessage1 extends AbstractTaggedIpmiPayload {
025
026    public static enum PrivilegeLookupMode implements Bits.Wrapper, Code.Wrapper {
027
028        /**
029         * Both the Requested Privilege Level and User Name are used to look up password/key.
030         * The BMC will search the user entries
031         * starting with USER ID 1 and use the first entry that matches the specified user
032         * name and has a Maximum Privilege Level that matches the Requested
033         * Privilege Level. This can be used in combination with ‘null’ user names to
034         * enable a “role only” login with a password that is just associated with the
035         * requested privilege level.
036         */
037        USERNAME_PRIVILEGE(new Bits(0, 1 << 4, 0)),
038        /**
039         * User Name alone is used to look up password/key.
040         * Privilege Level field acts as a ‘Maximum Requested Privilege Level’ as
041         * in IPMI v1.5.
042         * <ul>
043         * <li>If the Requested Privilege Level is greater than the privilege limit for the
044         * channel/user, the user will be allowed to connect but will be restricted to the
045         * channel/user privilege limit that was configured for the user.</li>
046         * <li>If the Requested Privilege Level is less than the channel/user privilege limit,
047         * the user will be allowed to connect and Request Privilege Level will become
048         * the effective privilege limit for the user. I.e. the user will not be able to raise
049         * their privilege level higher than the Requested Privilege Level.</li>
050         * </ul>
051         */
052        NAME_ONLY(new Bits(0, 1 << 4, 1 << 4));
053        public static final int MASK = 0x0F << 4;
054        private final Bits bits;
055
056        private PrivilegeLookupMode(@Nonnull Bits bits) {
057            this.bits = bits;
058        }
059
060        @Override
061        public Bits getBits() {
062            return bits;
063        }
064
065        @Override
066        public byte getCode() {
067            return UnsignedBytes.checkedCast(getBits().getByteValue());
068        }
069    }
070    private int systemSessionId;
071    private byte[] consoleRandom;   // length = 16
072    private RequestedMaximumPrivilegeLevel requestedMaximumPrivilegeLevel = RequestedMaximumPrivilegeLevel.ADMINISTRATOR;
073    private PrivilegeLookupMode privilegeLookupMode = PrivilegeLookupMode.USERNAME_PRIVILEGE;
074    private String username;
075
076    public IpmiRAKPMessage1() {
077    }
078
079    public IpmiRAKPMessage1(@Nonnull IpmiSession session) {
080        this.systemSessionId = session.getSystemSessionId();
081        byte[] tmp = new byte[16];
082        IpmiSessionManager.RANDOM.nextBytes(tmp);
083        this.consoleRandom = tmp;
084    }
085
086    @Override
087    public IpmiPayloadType getPayloadType() {
088        return IpmiPayloadType.RAKPMessage1;
089    }
090
091    @Override
092    public Class<? extends AbstractTaggedIpmiPayload> getRequestType() {
093        return IpmiRAKPMessage1.class;
094    }
095
096    @Override
097    public Class<? extends AbstractTaggedIpmiPayload> getResponseType() {
098        return IpmiRAKPMessage2.class;
099    }
100
101    @Override
102    public void apply(IpmiClientIpmiPayloadHandler handler, IpmiHandlerContext context, IpmiSession session) {
103        handler.handleRAKPMessage1(context, session, this);
104    }
105
106    @Override
107    public int getWireLength(IpmiPacketContext context) {
108        return 28 + ((username == null) ? 0 : username.length());
109    }
110
111    @Override
112    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
113        buffer.put(messageTag);
114        buffer.put(new byte[3]);    // reserved
115        toWireIntLE(buffer, systemSessionId);
116        buffer.put(consoleRandom);
117        buffer.put(Bits.toByte(requestedMaximumPrivilegeLevel, privilegeLookupMode));
118        buffer.putChar((char) 0); // reserved
119        if (username != null) {
120            byte[] usernameBytes = username.getBytes(Charsets.ISO_8859_1);
121            buffer.put(UnsignedBytes.checkedCast(usernameBytes.length));    // Max is 0x10.
122            buffer.put(usernameBytes);
123        } else {
124            buffer.put((byte) 0);
125        }
126    }
127
128    @Override
129    protected void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
130        messageTag = buffer.get();
131        assertWireBytesZero(buffer, 3);
132        systemSessionId = fromWireIntLE(buffer);
133        consoleRandom = readBytes(buffer, 16);
134        byte requestedMaximumPrivilegeLevelByte = buffer.get();
135        requestedMaximumPrivilegeLevel = Code.fromByte(RequestedMaximumPrivilegeLevel.class, (byte) (requestedMaximumPrivilegeLevelByte & RequestedMaximumPrivilegeLevel.MASK));
136        privilegeLookupMode = Code.fromByte(PrivilegeLookupMode.class, (byte) (requestedMaximumPrivilegeLevelByte & PrivilegeLookupMode.MASK));
137        assertWireBytesZero(buffer, 2);
138        int usernameLength = UnsignedBytes.toInt(buffer.get());
139        if (usernameLength > 0) {
140            byte[] usernameBytes = readBytes(buffer, usernameLength);
141            username = new String(usernameBytes, Charsets.ISO_8859_1);
142        } else {
143            username = null;
144        }
145    }
146
147    @Override
148    public void toStringBuilder(StringBuilder buf, int depth) {
149        appendHeader(buf, depth, getClass().getSimpleName());
150        depth++;
151        appendValue(buf, depth, "MessageTag", "0x" + UnsignedBytes.toString(messageTag, 16));
152        appendValue(buf, depth, "SystemSessionId", "0x" + Integer.toHexString(systemSessionId));
153        appendValue(buf, depth, "ConsoleRandom", toHexString(consoleRandom));
154        appendValue(buf, depth, "RequestedMaximumPrivilegeLevel", requestedMaximumPrivilegeLevel);
155        appendValue(buf, depth, "PrivilegeLookupMode", privilegeLookupMode);
156        appendValue(buf, depth, "Username", username);
157    }
158}
