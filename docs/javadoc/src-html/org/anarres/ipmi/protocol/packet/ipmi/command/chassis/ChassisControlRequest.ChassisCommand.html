005package org.anarres.ipmi.protocol.packet.ipmi.command.chassis;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.common.Code;
014import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
015import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiRequest;
016
017/**
018 * [IPMI2] Section 28.3, table 28-4, page 390.
019 *
020 * @author shevek
021 */
022public class ChassisControlRequest extends AbstractIpmiRequest {
023
024    /**
025     * [IPMI2] Section 28.3, table 28-4, page 390.
026     */
027    public static enum ChassisCommand implements Code.DescriptiveWrapper {
028
029        /** Force system into soft off (S4/S45) state.  This is for 'emergency'
030         * management power down actions. The command does not initiate a clean
031         * shut-down of the operating system prior to powering down the system. */
032        PowerDown(0x00, "Power Down"),
033        PowerUp(0x01, "Power Up"),
034        /**
035         * This command provides a power off interval of at least 1 second
036         * following the deassertion of the system's
037         * POWERGOOD status from the main power subsystem. It is
038         * recommended that no action occur if system power is off (S4/S5)
039         * when this action is selected, and that a D5h 'Request parameter(s)
040         * not supported in present state.' error completion code be returned.
041         * Note that some implementations may cause a system power up if a
042         * power cycle operation is selected when system power is down. For
043         * consistency of operation, it is recommended that system
044         * management software first check the system power state before
045         * issuing a power cycle, and only issue the command if system
046         * power is ON or in a lower sleep state than S4/S5. */
047        PowerCycle(0x02, "Power Cycle"),
048        /** In some implementations, the BMC may not know
049         * whether a reset will cause any particular effect and will pulse the
050         * system reset signal regardless of power state. If the implementation
051         * can tell that no action will occur if a reset is delivered in a given
052         * power state, then it is recommended (but still optional) that a D5h
053         * 'Request parameter(s) not supported in present state.' error
054         * completion code be returned. */
055        HardReset(0x03, "Hard Reset"),
056        /** Pulse a version of a diagnostic
057         * interrupt that goes directly to the processor(s). This is typically used
058         * to cause the operating system to do a diagnostic dump (OS
059         * dependent). The interrupt is commonly an NMI on IA-32 systems
060         * and an INIT on Intel(r) Itanium(tm) processor based systems. */
061        PulseDiagnosticInterrupt(0x04, "Pulse Diagnostic Interrupt"),
062        ACPIOverTemperatureShutdown(0x05, "Initiate a soft-shutdown of OS via ACPI by emulating a fatal overtemperature");
063        private final byte code;
064        private final String description;
065        /* pp */ private ChassisCommand(@Nonnegative int code, @Nonnull String description) {
066            this.code = UnsignedBytes.checkedCast(code);
067            this.description = description;
068        }
069
070        @Override
071        public byte getCode() {
072            return code;
073        }
074
075        @Override
076        public String getDescription() {
077            return description;
078        }
079
080        @Override
081        public String toString() {
082            return Code.Utils.toString(this);
083        }
084    }
085    public ChassisCommand chassisCommand;
086
087    @Override
088    public IpmiCommandName getCommandName() {
089        return IpmiCommandName.ChassisControl;
090    }
091
092    @Override
093    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
094        handler.handleChassisControlRequest(context, this);
095    }
096
097    @Override
098    protected int getDataWireLength() {
099        return 1;
100    }
101
102    @Override
103    protected void toWireData(ByteBuffer buffer) {
104        buffer.put(chassisCommand.getCode());
105    }
106
107    @Override
108    protected void fromWireData(ByteBuffer buffer) {
109        chassisCommand = Code.fromBuffer(ChassisCommand.class, buffer);
110    }
111
112    @Override
113    public void toStringBuilder(StringBuilder buf, int depth) {
114        super.toStringBuilder(buf, depth);
115        appendValue(buf, depth, "ChassisCommand", chassisCommand);
116    }
117}
