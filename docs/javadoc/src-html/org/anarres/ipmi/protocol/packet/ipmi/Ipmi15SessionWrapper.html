005package org.anarres.ipmi.protocol.packet.ipmi;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import org.anarres.ipmi.protocol.packet.common.AbstractWireable;
010import org.anarres.ipmi.protocol.packet.common.Code;
011import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayload;
012import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayloadType;
013import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * [IPMI2] Section 13.6 pages 133-134, column 1.
019 *
020 * @author shevek
021 */
022public class Ipmi15SessionWrapper extends AbstractIpmiSessionWrapper {
023
024    private static final Logger LOG = LoggerFactory.getLogger(Ipmi15SessionWrapper.class);
025    // private IpmiSessionAuthenticationType authenticationType = IpmiSessionAuthenticationType.NONE;
026    // private int ipmiSessionSequenceNumber;
027    // private int ipmiSessionId;
028    // private byte[] ipmiMessageAuthenticationCode;   // 16 bytes
029    private IpmiSessionAuthenticationType authenticationType = IpmiSessionAuthenticationType.NONE;
030
031    @Override
032    public boolean isEncrypted() {
033        return false;
034    }
035
036    @Override
037    public boolean isAuthenticated() {
038        return !IpmiSessionAuthenticationType.NONE.equals(authenticationType);
039    }
040
041    // @Override
042    // public int getIpmiSessionId() { return ipmiSessionId; }
043    // @Override
044    // public int getIpmiSessionSequenceNumber() { return ipmiSessionSequenceNumber; }
045    @Override
046    public int getWireLength(IpmiPacketContext context) {
047        return 1 // authenticationType
048                + 4 // ipmiSessionSequenceNumber
049                + 4 // ipmiSessionId
050                + (IpmiSessionAuthenticationType.NONE.equals(authenticationType) ? 0 : 16)
051                + 1 // payloadLength
052                + getIpmiPayload().getWireLength(context);
053    }
054
055    /** Sequence number handling: [IPMI2] Section 6.12.8, page 58. */
056    @Override
057    public void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
058        // Page 133
059        buffer.put(authenticationType.getCode());
060        buffer.putInt(getIpmiSessionSequenceNumber());
061        buffer.putInt(getIpmiSessionId());
062        if (authenticationType != IpmiSessionAuthenticationType.NONE) {
063            byte[] ipmiMessageAuthenticationCode = null;    // new byte 16
064            buffer.put(ipmiMessageAuthenticationCode);
065        }
066        // Page 134
067        IpmiPayload payload = getIpmiPayload();
068        int payloadLength = payload.getWireLength(context);
069        // LOG.info("payload=" + payload);
070        // LOG.info("payloadLength=" + payloadLength);
071        buffer.put(UnsignedBytes.checkedCast(payloadLength));
072
073        payload.toWire(context, buffer);
074    }
075
076    @Override
077    public void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
078        IpmiSessionAuthenticationType authenticationType = Code.fromBuffer(IpmiSessionAuthenticationType.class, buffer);
079        setIpmiSessionSequenceNumber(buffer.getInt());
080        setIpmiSessionId(buffer.getInt());
081
082        if (authenticationType != IpmiSessionAuthenticationType.NONE) {
083            byte[] ipmiMessageAuthenticationCode = AbstractWireable.readBytes(buffer, 16);
084        }
085        byte payloadLength = buffer.get();
086
087        ByteBuffer payloadBuffer = buffer.duplicate();
088        payloadBuffer.limit(payloadBuffer.position() + UnsignedBytes.toInt(payloadLength));
089        buffer.position(payloadBuffer.limit());
090
091        IpmiPayload payload = newPayload(payloadBuffer, IpmiPayloadType.IPMI);
092        payload.fromWire(context, payloadBuffer);
093        setIpmiPayload(payload);
094
095        // assert payloadLength == header.getWireLength() + payload.getWireLength();
096    }
097}
