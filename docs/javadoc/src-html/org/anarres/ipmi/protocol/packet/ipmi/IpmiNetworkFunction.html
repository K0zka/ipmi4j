005package org.anarres.ipmi.protocol.packet.ipmi;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import org.anarres.ipmi.protocol.packet.common.Code;
010
011/**
012 * [IPMI2] Section 5.1, Pages 40-41.
013 * 
014 * Called a NetFn in the documentation.
015 * Even numbers are requests. Odd numbers are responses.
016 *
017 * @author shevek
018 */
019public enum IpmiNetworkFunction implements Code.Wrapper {
020
021    /** Chassis Device Requests and Responses. */
022    Chassis(0x00),
023    /** Bridge Requests and Responses. */
024    Bridge(0x02),
025    /** Sensor and Event Requests and Responses. */
026    Sensor(0x04),
027    /** Application Requests and Responses. */
028    App(0x06),
029    /** Firmware Transfer Requests and Responses. */
030    Firmware(0x08),
031    /** Non-volatile Storage Requests and Responses. */
032    Storage(0x0A),
033    /** Media-specific Configuration and Control. */
034    Transport(0x0C),
035    /** Non-IPMI Group Requests and Responses. */
036    GroupExtension(0x2C),
037    /** OEM/Non- IPMI Group Requests and Response. */
038    OEM_Group(0x2E),
039    ControllerSpecific0(0x30),
040    ControllerSpecific1(0x32),
041    ControllerSpecific2(0x34),
042    ControllerSpecific3(0x36),
043    ControllerSpecific4(0x38),
044    ControllerSpecific5(0x3A),
045    ControllerSpecific6(0x3C),
046    ControllerSpecific7(0x3E);
047    private final byte code;
048    // private Bits bits;
049
050    private IpmiNetworkFunction(@Nonnegative int code) {
051        this.code = UnsignedBytes.checkedCast(code);
052        // this.bits = new Bits(0, 0xFC, code << 2);
053    }
054
055    @Override
056    public byte getCode() {
057        return code;
058    }
059}
