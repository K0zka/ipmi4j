005package org.anarres.ipmi.protocol.packet.asf;
006
007import java.nio.ByteBuffer;
008import javax.annotation.Nonnull;
009import org.anarres.ipmi.protocol.client.visitor.IpmiClientAsfMessageHandler;
010import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
011import org.anarres.ipmi.protocol.packet.common.Code;
012
013/**
014 * RAKPMessage3.
015 * 
016 * http://www.dmtf.org/sites/default/files/standards/documents/DSP0136.pdf
017 * http://www.dmtf.org/standards/asf
018 * Section 3.2.4.15 page 43.
019 *
020 * @author shevek
021 */
022public class AsfRAKPMessage3Data extends AbstractAsfData {
023
024    private AsfRsspSessionStatus status = AsfRsspSessionStatus.NO_ERROR;
025    private int clientSessionId;
026
027    @Nonnull
028    public AsfRsspSessionStatus getStatus() {
029        return status;
030    }
031
032    @Nonnull
033    public AsfRAKPMessage3Data withStatus(@Nonnull AsfRsspSessionStatus status) {
034        this.status = status;
035        return this;
036    }
037
038    public int getClientSessionId() {
039        return clientSessionId;
040    }
041
042    @Nonnull
043    public AsfRAKPMessage3Data withClientSessionId(int clientSessionId) {
044        this.clientSessionId = clientSessionId;
045        return this;
046    }
047
048    @Override
049    public AsfRmcpMessageType getMessageType() {
050        return AsfRmcpMessageType.RAKPMessage3;
051    }
052
053    @Override
054    public void apply(IpmiClientAsfMessageHandler handler, IpmiHandlerContext context) {
055        handler.handleAsfRAKPMessage3Data(context, this);
056    }
057
058    @Override
059    protected int getDataWireLength() {
060        if (!AsfRsspSessionStatus.NO_ERROR.equals(getStatus()))
061            return 8;
062        byte[] integrityData = null;
063        return 8 + integrityData.length;
064    }
065
066    @Override
067    protected void toWireData(ByteBuffer buffer) {
068        AsfRsspSessionStatus s = getStatus();
069        buffer.put(s.getCode());
070        buffer.put(new byte[3]);  // Reserved
071        buffer.putInt(getClientSessionId());
072        if (!AsfRsspSessionStatus.NO_ERROR.equals(s))
073            return;
074        byte[] integrityData = null;
075        buffer.put(integrityData);
076    }
077
078    @Override
079    protected void fromWireData(ByteBuffer buffer) {
080        AsfRsspSessionStatus s = Code.fromBuffer(AsfRsspSessionStatus.class, buffer);
081        withStatus(s);
082        assertWireBytesZero(buffer, 3);
083        withClientSessionId(buffer.getInt());
084        if (!AsfRsspSessionStatus.NO_ERROR.equals(s))
085            return;
086        byte[] integrityData = null;
087        buffer.get(integrityData);
088    }
089}
