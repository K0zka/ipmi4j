005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.base.Preconditions;
008import java.nio.ByteBuffer;
009import java.util.UUID;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientAsfMessageHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.common.Code;
014
015/**
016 * RAKPMessage2.
017 * 
018 * http://www.dmtf.org/sites/default/files/standards/documents/DSP0136.pdf
019 * http://www.dmtf.org/standards/asf
020 * Section 3.2.4.14 page 42.
021 *
022 * @author shevek
023 */
024public class AsfRAKPMessage2Data extends AbstractAsfData {
025
026    private AsfRsspSessionStatus status = AsfRsspSessionStatus.NO_ERROR;
027    private int consoleSessionId;
028    private byte[] clientRandom;      // length = 16
029    private UUID clientGuid;
030
031    @Nonnull
032    public AsfRsspSessionStatus getStatus() {
033        return status;
034    }
035
036    @Nonnull
037    public AsfRAKPMessage2Data withStatus(@Nonnull AsfRsspSessionStatus status) {
038        this.status = status;
039        return this;
040    }
041
042    public int getConsoleSessionId() {
043        return consoleSessionId;
044    }
045
046    @Nonnull
047    public AsfRAKPMessage2Data withConsoleSessionId(int consoleSessionId) {
048        this.consoleSessionId = consoleSessionId;
049        return this;
050    }
051
052    @Nonnull
053    public byte[] getClientRandom() {
054        return clientRandom;
055    }
056
057    @Nonnull
058    public AsfRAKPMessage2Data withClientRandom(@Nonnull byte[] clientRandom) {
059        Preconditions.checkNotNull(clientRandom, "Random must be non-null.");
060        Preconditions.checkArgument(clientRandom.length == 16, "Random length must be 16.");
061        this.clientRandom = clientRandom;
062        return this;
063    }
064
065    public UUID getClientGuid() {
066        return clientGuid;
067    }
068
069    @Nonnull
070    public AsfRAKPMessage2Data withClientGuid(UUID clientGuid) {
071        this.clientGuid = clientGuid;
072        return this;
073    }
074
075    @Override
076    public AsfRmcpMessageType getMessageType() {
077        return AsfRmcpMessageType.RAKPMessage2;
078    }
079
080    @Override
081    public void apply(IpmiClientAsfMessageHandler handler, IpmiHandlerContext context) {
082        handler.handleAsfRAKPMessage2Data(context, this);
083    }
084
085    @Override
086    protected int getDataWireLength() {
087        if (!AsfRsspSessionStatus.NO_ERROR.equals(getStatus()))
088            return 8;
089        byte[] integrityData = null;
090        return 40 + integrityData.length;
091    }
092
093    @Override
094    protected void toWireData(ByteBuffer buffer) {
095        AsfRsspSessionStatus s = getStatus();
096        buffer.put(s.getCode());
097        buffer.put(new byte[3]);  // Reserved
098        buffer.putInt(getConsoleSessionId());
099        if (!AsfRsspSessionStatus.NO_ERROR.equals(s))
100            return;
101        buffer.put(getClientRandom());
102        // 16 byte GUID
103        UUID g = getClientGuid();
104        buffer.putLong(g.getMostSignificantBits());
105        buffer.putLong(g.getLeastSignificantBits());
106        byte[] integrityData = null;
107        buffer.put(integrityData);
108    }
109
110    @Override
111    protected void fromWireData(ByteBuffer buffer) {
112        AsfRsspSessionStatus s = Code.fromBuffer(AsfRsspSessionStatus.class, buffer);
113        withStatus(s);
114        assertWireBytesZero(buffer, 3);
115        withConsoleSessionId(buffer.getInt());
116        if (!AsfRsspSessionStatus.NO_ERROR.equals(s))
117            return;
118        withClientRandom(readBytes(buffer, 16));
119        // This is sequenced in Java, right?
120        withClientGuid(new UUID(buffer.getLong(), buffer.getLong()));
121        byte[] integrityData = null;
122        buffer.get(integrityData);
123    }
124}
