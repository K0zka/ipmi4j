039    private static final IpmiSessionAuthenticationType AUTHENTICATION_TYPE = IpmiSessionAuthenticationType.RMCPP;

055    @Nonnull
056    public IpmiConfidentialityAlgorithm getConfidentialityAlgorithm(@CheckForNull IpmiSession session) {
057        if (!isEncrypted())
058            return IpmiConfidentialityAlgorithm.NONE;
059        if (getIpmiSessionId() == 0)
060            return IpmiConfidentialityAlgorithm.NONE;
061        if (session == null)
062            throw new IllegalStateException("Confidentiality requested, but cannot find session " + Integer.toHexString(getIpmiSessionId()));
063        return session.getConfidentialityAlgorithm();
064    }

075    @Nonnull
076    public IpmiAuthenticationAlgorithm getAuthenticationAlgorithm(@CheckForNull IpmiSession session) {
077        if (!isEncrypted())
078            return IpmiAuthenticationAlgorithm.RAKP_NONE;
079        if (getIpmiSessionId() == 0)
080            return IpmiAuthenticationAlgorithm.RAKP_NONE;
081        if (session == null)
082            throw new IllegalStateException("Authentication requested, but cannot find session " + Integer.toHexString(getIpmiSessionId()));
083        return session.getAuthenticationAlgorithm();
084    }

086    @Override
087    public int getWireLength(IpmiPacketContext context) {
088        try {
089            @CheckForNull
090            IpmiSession session = context.getIpmiSession(getIpmiSessionId());
091
092            IpmiConfidentialityAlgorithm confidentialityAlgorithm = getConfidentialityAlgorithm(session);
093            IpmiIntegrityAlgorithm integrityAlgorithm = getIntegrityAlgorithm(session);
094
095            IpmiPayload payload = getIpmiPayload();
096
097            boolean oem = IpmiPayloadType.OEM_EXPLICIT.equals(payload.getPayloadType());
098            int unencryptedLength = payload.getWireLength(context);
099            int encryptedLength = confidentialityAlgorithm.getEncryptedLength(session, unencryptedLength);
100            int integrityLength = (session == null)
101                    ? 0
102                    : IntegrityPad.PAD(encryptedLength).length
103                    + 1 // pad length
104                    + 1 // next header
105                    + integrityAlgorithm.getMacLength();
106            return 1 // authenticationType
107                    + 1 // payloadType
108                    + (oem ? 4 : 0) // oemEnterpriseNumber
109                    + (oem ? 2 : 0) // oemPayloadId
110                    + 4 // ipmiSessionId
111                    + 4 // ipmiSessionSequenceNumber
112                    + 2 // payloadLength
113                    + encryptedLength
114                    + integrityLength;
115        } catch (NoSuchAlgorithmException e) {
116            throw Throwables.propagate(e);
117        }
118    }

120    /** Sequence number handling: [IPMI2] Section 6.12.13, page 59. */
121    @Override
122    public void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
123        try {
124            @CheckForNull
125            IpmiSession session = context.getIpmiSession(getIpmiSessionId());
126
127            IpmiConfidentialityAlgorithm confidentialityAlgorithm = getConfidentialityAlgorithm(session);
128            IpmiAuthenticationAlgorithm authenticationAlgorithm = getAuthenticationAlgorithm(session);
129            IpmiIntegrityAlgorithm integrityAlgorithm = getIntegrityAlgorithm(session);
130
131            boolean encrypted = !IpmiConfidentialityAlgorithm.NONE.equals(confidentialityAlgorithm);
132            boolean authenticated = !IpmiAuthenticationAlgorithm.RAKP_NONE.equals(authenticationAlgorithm);
133
134            IpmiPayload payload = getIpmiPayload();
135
136            // Page 133
137            buffer.put(AUTHENTICATION_TYPE.getCode());
138            byte payloadTypeByte = payload.getPayloadType().getCode();
139            payloadTypeByte = AbstractIpmiCommand.setBit(payloadTypeByte, 7, encrypted);
140            payloadTypeByte = AbstractIpmiCommand.setBit(payloadTypeByte, 6, authenticated);
141            buffer.put(payloadTypeByte);
142            if (IpmiPayloadType.OEM_EXPLICIT.equals(payload.getPayloadType())) {
143                OemExplicit oemPayload = (OemExplicit) payload;
144                buffer.putInt(oemPayload.getOemEnterpriseNumber() << Byte.SIZE);
145                buffer.putChar(oemPayload.getOemPayloadId());
146            }
147            toWireIntLE(buffer, getIpmiSessionId());
148            toWireIntLE(buffer, getIpmiSessionSequenceNumber());
149
150            ByteBuffer integrityInput = buffer.duplicate();
151
152            // Page 134
153            // 2 byte payload length
154            int unencryptedLength = payload.getWireLength(context);
155            int encryptedLength = confidentialityAlgorithm.getEncryptedLength(session, unencryptedLength);
156            toWireCharLE(buffer, Chars.checkedCast(encryptedLength));
157
158            ENCRYPT:
159            {
160                if (encrypted) {
161                    ByteBuffer unencryptedBuffer = ByteBuffer.allocate(unencryptedLength);
162                    payload.toWire(context, unencryptedBuffer);
163                    unencryptedBuffer.flip();
164                    confidentialityAlgorithm.encrypt(session, buffer, unencryptedBuffer);
165                } else {
166                    // Avoidance of allocation optimization.
167                    payload.toWire(context, buffer);
168                }
169            }
170
171            SIGN:
172            if (!IpmiIntegrityAlgorithm.NONE.equals(integrityAlgorithm)) {
173                // Integrity padding.
174                byte[] pad = IntegrityPad.PAD(encryptedLength);
175                buffer.put(pad);
176                buffer.put((byte) pad.length);
177                buffer.put((byte) 0x07); // Reserved, [IPMI2] Page 134, next-header field.
178
179                integrityInput.limit(buffer.position());
180                byte[] integrityData = integrityAlgorithm.sign(session, integrityInput);
181                buffer.put(integrityData);
182            }
183        } catch (GeneralSecurityException e) {
184            throw Throwables.propagate(e);
185        }
186    }

188    @Override
189    public void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
190        try {
191            AbstractWireable.assertWireByte(buffer, AUTHENTICATION_TYPE.getCode(), "IPMI session authentication type");
192            byte payloadTypeByte = buffer.get();
193            encrypted = AbstractIpmiCommand.getBit(payloadTypeByte, 7);
194            authenticated = AbstractIpmiCommand.getBit(payloadTypeByte, 6);
195            IpmiPayloadType payloadType = Code.fromInt(IpmiPayloadType.class, payloadTypeByte & 0x3F);
196
197            int sessionId = fromWireIntLE(buffer);
198            setIpmiSessionId(sessionId);
199            setIpmiSessionSequenceNumber(fromWireIntLE(buffer));
200
201            IpmiSession session = context.getIpmiSession(sessionId);
202            IpmiAuthenticationAlgorithm authenticationAlgorithm = getAuthenticationAlgorithm(session);
203            IpmiConfidentialityAlgorithm confidentialityAlgorithm = getConfidentialityAlgorithm(session);
204            IpmiIntegrityAlgorithm integrityAlgorithm = getIntegrityAlgorithm(session);
205
206            ByteBuffer integrityInput = buffer.duplicate();
207
208            // TODO: Before calling payload.fromWire(), make sure to set the position and limit on the buffer.
209            int payloadLength = fromWireCharLE(buffer);
210            ByteBuffer encryptedBuffer = buffer.duplicate();
211            // LOG.info("position=" + encryptedBuffer.position());
212            // LOG.info("length=0x" + Integer.toHexString(payloadLength));
213            encryptedBuffer.limit(encryptedBuffer.position() + payloadLength);
214            buffer.position(encryptedBuffer.limit());
215
216            DECRYPT:
217            {
218                ByteBuffer unencryptedBuffer = isEncrypted()
219                        ? session.getConfidentialityAlgorithm().decrypt(session, encryptedBuffer)
220                        : encryptedBuffer;
221                IpmiPayload payload = newPayload(unencryptedBuffer, payloadType);
222                payload.fromWire(context, unencryptedBuffer);
223                setIpmiPayload(payload);
224
225                if (unencryptedBuffer.hasRemaining()) {
226                    throw new IllegalStateException(unencryptedBuffer.remaining()
227                            + " bytes remaining in buffer after decoding " + payload);
228                }
229            }
230
231            SIGN:
232            if (!IpmiIntegrityAlgorithm.NONE.equals(integrityAlgorithm)) {
233                int integrityPadLength = IntegrityPad.PAD(payloadLength).length;
234                // LOG.info("IntegrityPad length is " + integrityPadLength);
235                byte[] integrityPad = AbstractWireable.readBytes(buffer, integrityPadLength);
236                // TODO: Assert integrityPad all 0xFF.
237
238                AbstractWireable.assertWireByte(buffer, UnsignedBytes.checkedCast(integrityPadLength), "Integrity pad length");
239                AbstractWireable.assertWireByte(buffer, (byte) 0x07, "Next-header field");
240
241                integrityInput.limit(buffer.position());
242                byte[] integrityData = session.getIntegrityAlgorithm().sign(session, integrityInput);
243                // TODO: Assert integrityData equal to outstanding buffer data.
244            }
245        } catch (GeneralSecurityException e) {
246            throw Throwables.propagate(e);
247        }
248    }
