005package org.anarres.ipmi.protocol.packet.ipmi;
006
007import com.google.common.primitives.UnsignedBytes;
008import javax.annotation.Nonnegative;
009import org.anarres.ipmi.protocol.packet.common.Bits;
010import org.anarres.ipmi.protocol.packet.common.Code;
011
012/**
013 * [IPMI2] Section 6.3, table 6-1, page 48.
014 * [IPMI2] Section 22.13, table 22-15, page 283.
015 *
016 * @author shevek
017 */
018public enum IpmiChannelNumber implements Code.Wrapper {
019
020    PRIMARY_IPMB(0x0),
021    C1(0x1),
022    C2(0x2),
023    C3(0x3),
024    C4(0x4),
025    C5(0x5),
026    C6(0x6),
027    C7(0x7),
028    C8(0x8),
029    C9(0x9),
030    CA(0xA),
031    CB(0xB),
032    CURRENT(0xE),
033    /** The system interface? */
034    CF(0xF);
035    private final byte code;
036
037    private IpmiChannelNumber(@Nonnegative int code) {
038        this.code = Bits.validate(0xF, code);
039    }
040
041    @Override
042    public byte getCode() {
043        return code;
044    }
045
046    @Override
047    public String toString() {
048        return name() + "(" + UnsignedBytes.toString(getCode(), 16) + ")";
049    }
050}
