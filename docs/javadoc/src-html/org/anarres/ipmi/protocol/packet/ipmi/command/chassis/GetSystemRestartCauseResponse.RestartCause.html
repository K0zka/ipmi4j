005package org.anarres.ipmi.protocol.packet.ipmi.command.chassis;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.common.Code;
014import org.anarres.ipmi.protocol.packet.ipmi.IpmiChannelNumber;
015import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
016import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiResponse;
017
018/**
019 * [IPMI2] Section 28.2, table 28-3, page 389.
020 *
021 * @author shevek
022 */
023public class GetSystemRestartCauseResponse extends AbstractIpmiResponse {
024
025    public static enum RestartCause implements Code.DescriptiveWrapper {
026
027        Unknown(0x00, "unknown (system start/restart detected, but cause unknown)"),
028        ChassisControlCommand(0x01, "Chassis Control command"),
029        PushButtonReset(0x02, "reset via pushbutton"),
030        PushButtonPowerUp(0x03, "power-up via power pushbutton"),
031        WatchdogExpiration(0x04, "Watchdog expiration (see watchdog flags)"),
032        OEM(0x05, "OEM"),
033        AutomaticAlwaysRestore(0x06, "automatic power-up on AC being applied due to 'always restore' power restore policy "),
034        AutomaticRestorePrevious(0x07, "automatic power-up on AC being applied due to 'restore previous power state' power restore policy"),
035        PEFReset(0x08, "reset via PEF"),
036        PEFPowerCycle(0x09, "power-cycle via PEF"),
037        SoftReset(0x0A, "soft reset (e.g. CTRL-ALT-DEL)"),
038        RTCPowerUp(0x0B, "power-up via RTC (system real time clock) wakeup");
039        private final byte code;
040        private final String description;
041        /* pp */ private RestartCause(@Nonnegative int code, @Nonnull String description) {
042            this.code = UnsignedBytes.checkedCast(code);
043            this.description = description;
044        }
045
046        @Override
047        public byte getCode() {
048            return code;
049        }
050
051        @Override
052        public String getDescription() {
053            return description;
054        }
055
056        @Override
057        public String toString() {
058            return Code.Utils.toString(this);
059        }
060    }
061    public RestartCause restartCause;
062    public IpmiChannelNumber restartChannel;
063
064    @Override
065    public IpmiCommandName getCommandName() {
066        return IpmiCommandName.GetSystemRestartCause;
067    }
068
069    @Override
070    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
071        handler.handleGetSystemRestartCauseResponse(context, this);
072    }
073
074    @Override
075    protected int getResponseDataWireLength() {
076        return 3;
077    }
078
079    @Override
080    protected void toWireData(ByteBuffer buffer) {
081        if (toWireCompletionCode(buffer))
082            return;
083        buffer.put(restartCause.getCode());
084        buffer.put(restartChannel.getCode());
085    }
086
087    @Override
088    protected void fromWireData(ByteBuffer buffer) {
089        if (fromWireCompletionCode(buffer))
090            return;
091        byte tmp = buffer.get();
092        restartCause = Code.fromInt(RestartCause.class, tmp & 0xF);
093        restartChannel = Code.fromBuffer(IpmiChannelNumber.class, buffer);
094    }
095
096    @Override
097    public void toStringBuilder(StringBuilder buf, int depth) {
098        super.toStringBuilder(buf, depth);
099        appendValue(buf, depth, "RestartCause", restartCause);
100        appendValue(buf, depth, "RestartChannel", restartChannel);
101    }
102}
