022/**

027public enum IpmiConfidentialityAlgorithm implements IpmiAlgorithm {
028
029    NONE(0x00) {
030        @Override
031        public int getEncryptedLength(IpmiSession session, int unencryptedLength) {
032            return unencryptedLength;
033        }
034
035        @Override
036        public void encrypt(IpmiSession session, ByteBuffer out, ByteBuffer in) throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeyException, InvalidAlgorithmParameterException, ShortBufferException {
037            out.put(in);
038        }
039
040        @Override
041        public ByteBuffer decrypt(IpmiSession session, ByteBuffer in) throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeyException, InvalidAlgorithmParameterException, ShortBufferException {
042            return in;
043        }
044    },
045    /** [IPMI2] Section 13.29, table 13-20, page 160. */
046    AES_CBC_128(0x01) {
047        @Nonnegative
048        private int pad(@Nonnegative int dataLength) {
049            int m = dataLength % 16;
050            if (m == 0)
051                return 0;
052            return 16 - m;
053        }
054
055        @Override
056        public int getEncryptedLength(IpmiSession session, int unencryptedLength) {
057            return 16 // IV
058                    + unencryptedLength
059                    + pad(unencryptedLength) // trailer: encrypted by AES
060                    + 1;    // pad length
061        }
062
063        @Override
064        public void encrypt(IpmiSession session, ByteBuffer out, ByteBuffer in)
065                throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeyException, InvalidAlgorithmParameterException, ShortBufferException {
066            byte[] iv = session.newRandomSeed(16);
067            byte[] key = session.getAdditionalKey(2);
068            out.put(iv);
069            AES_CBC_128 cipher = new AES_CBC_128();
070            cipher.init(Cipher.Mode.ENCRYPT, key, iv);
071            cipher.update(in, out);
072
073            PAD:
074            {
075                int padLength = pad(in.remaining());
076                if (padLength > 0) {
077                    ByteBuffer pad = ByteBuffer.allocate(padLength);
078                    // Pad bytes shall start at 1 and have a monotonically increasing value.
079                    int i = 1;
080                    while (pad.hasRemaining())
081                        pad.put(UnsignedBytes.checkedCast(i++));
082                    cipher.update(pad, out);
083                }
084                out.put(UnsignedBytes.checkedCast(padLength));
085            }
086        }
087
088        @Override
089        public ByteBuffer decrypt(IpmiSession session, ByteBuffer in) throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeyException, InvalidAlgorithmParameterException, ShortBufferException {
090
091            int limit = in.limit() - 1;
092            int padLength = UnsignedBytes.toInt(in.get(limit));
093            in.limit(limit);
094
095            ByteBuffer payload = ByteBuffer.allocate(in.remaining());
096            byte[] iv = AbstractWireable.readBytes(in, 16);
097            byte[] key = session.getAdditionalKey(2);
098            AES_CBC_128 cipher = new AES_CBC_128();
099            cipher.init(Cipher.Mode.DECRYPT, key, iv);
100            cipher.update(payload, in); // TODO: Reads too much from buffer.
101
102            for (int i = 1; i <= padLength; i++)
103                if (UnsignedBytes.toInt(payload.get(payload.limit() - i)) != i)
104                    throw new IllegalArgumentException("Bad pad byte " + i);
105
106            payload.flip();
107            return payload;
108        }
109    },
110    /** [IPMI2] Section 13.30, table 13-21, page 161. */
111    xRC4_128(0x02) {
112        @Override
113        public int getEncryptedLength(IpmiSession session, int unencryptedLength) throws NoSuchAlgorithmException {
114            State state = session.getConfidentialityAlgorithmState();
115            // Either a data offset of zero (4 bytes) and a 16 byte IV, or a nonzero data offset.
116            if (state == null)
117                return 4 + 16 + unencryptedLength;
118            return 4 + unencryptedLength;
119        }
120    },
121    /** [IPMI2] Section 13.30, table 13-21, page 161. */
122    xRC4_40(0x03) {
123        @Override
124        public int getEncryptedLength(IpmiSession session, int unencryptedLength) {
125            State state = session.getConfidentialityAlgorithmState();
126            // Either a data offset of zero (4 bytes) and a 16 byte IV, or a nonzero data offset.
127            if (state == null)
128                return 4 + 16 + unencryptedLength;
129            return 4 + unencryptedLength;
130        }
131    },
132    OEM_30(0x30),
133    OEM_31(0x31),
134    OEM_32(0x32),
135    OEM_33(0x33),
136    OEM_34(0x34),
137    OEM_35(0x35),
138    OEM_36(0x36),
139    OEM_37(0x37),
140    OEM_38(0x38),
141    OEM_39(0x39),
142    OEM_3A(0x3A),
143    OEM_3B(0x3B),
144    OEM_3C(0x3C),
145    OEM_3D(0x3D),
146    OEM_3E(0x3E),
147    OEM_3F(0x3F);

150    }

156    public static final byte PAYLOAD_TYPE = 2;
157    private final byte code;
158
159    private IpmiConfidentialityAlgorithm(@Nonnegative int code) {
160        this.code = UnsignedBytes.checkedCast(code);
161    }
162
163    @Override
164    public byte getPayloadType() {
165        return PAYLOAD_TYPE;
166    }
167
168    @Override
169    public byte getCode() {
170        return code;
171    }
172
173    @Nonnegative
174    public int getEncryptedLength(@CheckForNull IpmiSession session, @Nonnegative int unencryptedLength)
175            throws NoSuchAlgorithmException {
176        throw new NoSuchAlgorithmException("Unsupported confidentiality algorithm " + this);
177    }
178
179    public void encrypt(@CheckForNull IpmiSession session, @Nonnull ByteBuffer out, @Nonnull ByteBuffer in)
180            throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeyException, InvalidAlgorithmParameterException, ShortBufferException {
181        throw new NoSuchAlgorithmException("Unsupported confidentiality algorithm " + this);
182    }
183
184    /** Called with a buffer of payload only. Will consume its entire buffer. */
185    @Nonnull
186    public ByteBuffer decrypt(@CheckForNull IpmiSession session, @Nonnull ByteBuffer in)
187            throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeyException, InvalidAlgorithmParameterException, ShortBufferException {
188        throw new NoSuchAlgorithmException("Unsupported confidentiality algorithm " + this);
189    }
