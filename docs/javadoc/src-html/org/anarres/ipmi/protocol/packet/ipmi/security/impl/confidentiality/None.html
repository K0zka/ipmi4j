005package org.anarres.ipmi.protocol.packet.ipmi.security.impl.confidentiality;
006
007import java.nio.ByteBuffer;
008import java.security.InvalidAlgorithmParameterException;
009import java.security.InvalidKeyException;
010import javax.crypto.ShortBufferException;
011import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiConfidentialityAlgorithm;
012
013/**
014 * [IPMI2] Section 13.28.5, page 159.
015 *
016 * @author shevek
017 */
018public class None implements Cipher {
019
020    @Override
021    public IpmiConfidentialityAlgorithm getName() {
022        return IpmiConfidentialityAlgorithm.NONE;
023    }
024
025    @Override
026    public int getIVSize() {
027        return 0;
028    }
029
030    @Override
031    public int getBlockSize() {
032        return 1;
033    }
034
035    @Override
036    public void init(Mode mode, byte[] key, byte[] iv) throws InvalidKeyException, InvalidAlgorithmParameterException {
037    }
038
039    @Override
040    public int update(ByteBuffer input, ByteBuffer output) throws ShortBufferException {
041        int remaining = input.remaining();
042        output.put(input);
043        return remaining;
044    }
045}
