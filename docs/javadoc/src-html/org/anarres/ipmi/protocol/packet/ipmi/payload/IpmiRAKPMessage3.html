005package org.anarres.ipmi.protocol.packet.ipmi.payload;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiPayloadHandler;
010import org.anarres.ipmi.protocol.packet.asf.AsfRsspSessionStatus;
011import org.anarres.ipmi.protocol.packet.common.Code;
012import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
013import org.anarres.ipmi.protocol.client.session.IpmiSession;
014import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
015
016/**
017 * [IPMI2] Section 13.22 page 152.
018 *
019 * @author shevek
020 */
021public class IpmiRAKPMessage3 extends AbstractTaggedIpmiPayload {
022
023    private AsfRsspSessionStatus statusCode;
024    private int systemSessionId;
025    private byte[] keyExchangeAuthenticationCode;
026
027    @Override
028    public IpmiPayloadType getPayloadType() {
029        return IpmiPayloadType.RAKPMessage3;
030    }
031
032    @Override
033    public Class<? extends AbstractTaggedIpmiPayload> getRequestType() {
034        return IpmiRAKPMessage3.class;
035    }
036
037    @Override
038    public Class<? extends AbstractTaggedIpmiPayload> getResponseType() {
039        return IpmiRAKPMessage4.class;
040    }
041
042    @Override
043    public void apply(IpmiClientIpmiPayloadHandler handler, IpmiHandlerContext context, IpmiSession session) {
044        handler.handleRAKPMessage3(context, session, this);
045    }
046
047    @Override
048    public int getWireLength(IpmiPacketContext context) {
049        return 8 + keyExchangeAuthenticationCode.length;
050    }
051
052    @Override
053    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
054        buffer.put(messageTag);
055        buffer.put(statusCode.getCode());
056        buffer.putChar((char) 0);    // reserved
057        toWireIntLE(buffer, systemSessionId);
058        buffer.put(keyExchangeAuthenticationCode);
059    }
060
061    @Override
062    protected void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
063        messageTag = buffer.get();
064        statusCode = Code.fromBuffer(AsfRsspSessionStatus.class, buffer);
065        assertWireBytesZero(buffer, 2);
066        systemSessionId = fromWireIntLE(buffer);
067        keyExchangeAuthenticationCode = readBytes(buffer, buffer.remaining());
068    }
069
070    @Override
071    public void toStringBuilder(StringBuilder buf, int depth) {
072        appendHeader(buf, depth, getClass().getSimpleName());
073        depth++;
074        appendValue(buf, depth, "MessageTag", "0x" + UnsignedBytes.toString(messageTag, 16));
075        appendValue(buf, depth, "StatusCode", statusCode);
076        appendValue(buf, depth, "SystemSessionId", "0x" + Integer.toHexString(systemSessionId));
077        appendValue(buf, depth, "KeyExchangeAuthenticationCode", toHexString(keyExchangeAuthenticationCode));
078    }
079}
