001package org.anarres.ipmi.protocol.packet.rmcp;
002
003import java.nio.ByteBuffer;
004import javax.annotation.Nonnull;
005import org.anarres.ipmi.protocol.packet.ipmi.security.impl.integrity.IntegrityPad;
006import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
007
008/**
009 * RMCP Security Extensions Protocol (RSP).
010 * 
011 * [ASF] Section 3.2.3 pages 23-25.
012 * 
013 * Uses UDP port 0x0298.
014 *
015 * @author shevek
016 */
017public class RspRmcpPacket extends AbstractPacket {
018
019    public static final int PORT = 0x298;
020    private int sessionId;
021    private int sessionSequenceNumber;
022    // For this specification, the mandatory-to-implement integrity algorithm is HMAC-SHA1-96 defined in [RFC2404]. 
023    private byte[] integrityData;
024
025    public int getSessionId() {
026        return sessionId;
027    }
028
029    @Nonnull
030    public RspRmcpPacket withSessionId(int sessionId) {
031        this.sessionId = sessionId;
032        return this;
033    }
034
035    public int getSessionSequenceNumber() {
036        return sessionSequenceNumber;
037    }
038
039    @Nonnull
040    public RspRmcpPacket withSessionSequenceNumber(int sessionSequenceNumber) {
041        this.sessionSequenceNumber = sessionSequenceNumber;
042        return this;
043    }
044
045    public byte[] getIntegrityData() {
046        return integrityData;
047    }
048
049    @Nonnull
050    public RspRmcpPacket withIntegrityData(byte[] integrityData) {
051        this.integrityData = integrityData;
052        return this;
053    }
054
055    @Override
056    public int getWireLength(IpmiPacketContext context) {
057        int length = 4 // sessionId
058                + 4 // sessionSequenceNumber
059                + getRawWireLength(context);
060        return length
061                + IntegrityPad.PAD(length).length // Padding
062                + 1 // Padding length
063                + 1 // "Next header" length
064                + integrityData.length;
065    }
066
067    @Override
068    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
069        int start = buffer.position();
070        buffer.putInt(sessionId);
071        buffer.putInt(sessionSequenceNumber);
072        toWireRaw(context, buffer);
073        if (sessionId != 0) {   // Page 24: Unsecured data.
074            // Padding aligns us to a DWORD, which is 4 bytes.
075            int length = buffer.position() - start;
076            byte[] pad = IntegrityPad.PAD(length);
077            buffer.put(pad);
078            buffer.put((byte) pad.length);
079            buffer.put(getVersion().getCode());
080            buffer.put(integrityData);
081        }
082    }
083
084    @Override
085    protected final void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
086        int start = buffer.position();
087        int sessionId = buffer.getInt();
088        withSessionId(sessionId);
089        withSessionSequenceNumber(buffer.getInt());
090        fromWireRaw(context, buffer);
091        if (sessionId != 0) {   // Page 24: Unsecured data.
092            int length = buffer.position() - start;
093            byte[] pad = IntegrityPad.PAD(length);
094            readBytes(buffer, pad.length);  // TODO: Write a temporary.
095            assertWireByte(buffer, (byte) pad.length, "padding length");
096            assertWireByte(buffer, getVersion().getCode(), "header version");
097            withIntegrityData(readBytes(buffer, buffer.remaining()));
098        }
099    }
100}
