005package org.anarres.ipmi.protocol.packet.ipmi.command.sel;
006
007import java.nio.ByteBuffer;
008import java.util.EnumSet;
009import java.util.Set;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.common.Bits;
014import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
015import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiResponse;
016import org.anarres.ipmi.protocol.packet.ipmi.command.sdr.GetSDRRepositoryInfoResponse;
017
018/**
019 * [IPMI2] Section 31.2, table 31-2, page 423.
020 *
021 * @see GetSDRRepositoryInfoResponse
022 * @author shevek
023 */
024public class GetSELInfoResponse extends AbstractIpmiResponse {
025
026    public static enum SupportedOperation implements Bits.Wrapper {
027
028        OverflowFlag(Bits.forBitIndex(0, 7)),
029        DeleteSELCommandSupported(Bits.forBitIndex(0, 3)),
030        PartialAddSELEntryCommandSupported(Bits.forBitIndex(0, 2)),
031        ReserveSELCommandSupported(Bits.forBitIndex(0, 1)),
032        GetSELAllocationInformationCommandSupported(Bits.forBitIndex(0, 0));
033        private final Bits bits;
034
035        /* pp */ SupportedOperation(@Nonnull Bits bits) {
036            this.bits = bits;
037        }
038
039        @Override
040        public Bits getBits() {
041            return bits;
042        }
043    }
044    public char recordCount;
045    public char freeSpace;
046    public int mostRecentAdditionTimestamp;
047    public int mostRecentDeletionTimestamp;
048    public Set<SupportedOperation> supportedOperations = EnumSet.noneOf(SupportedOperation.class);
049
050    @Override
051    public IpmiCommandName getCommandName() {
052        return IpmiCommandName.GetSELInfo;
053    }
054
055    @Override
056    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
057        handler.handleGetSELInfoResponse(context, this);
058    }
059
060    @Override
061    protected int getResponseDataWireLength() {
062        return 15;
063    }
064
065    @Override
066    protected void toWireData(ByteBuffer buffer) {
067        if (toWireCompletionCode(buffer))
068            return;
069        buffer.put((byte) 0x51);
070        toWireCharLE(buffer, recordCount);
071        toWireCharLE(buffer, freeSpace);
072        toWireIntLE(buffer, mostRecentAdditionTimestamp);
073        toWireIntLE(buffer, mostRecentDeletionTimestamp);
074        buffer.put(Bits.toByte(supportedOperations));
075    }
076
077    @Override
078    protected void fromWireData(ByteBuffer buffer) {
079        if (fromWireCompletionCode(buffer))
080            return;
081        assertWireByte(buffer, (byte) 0x51, "SDR Version");
082        recordCount = fromWireCharLE(buffer);
083        freeSpace = fromWireCharLE(buffer);
084        mostRecentAdditionTimestamp = fromWireIntLE(buffer);
085        mostRecentDeletionTimestamp = fromWireIntLE(buffer);
086        supportedOperations = Bits.fromBuffer(SupportedOperation.class, buffer, 1);
087    }
088
089    @Override
090    public void toStringBuilder(StringBuilder buf, int depth) {
091        super.toStringBuilder(buf, depth);
092        appendValue(buf, depth, "RecordCount", (int) recordCount);
093        appendValue(buf, depth, "FreeSpace", "0x" + Integer.toHexString(freeSpace));
094        appendValue(buf, depth, "MostRecentAdditionTimestamp", "0x" + Integer.toHexString(mostRecentAdditionTimestamp));
095        appendValue(buf, depth, "MostRecentDeletionTimestamp", "0x" + Integer.toHexString(mostRecentDeletionTimestamp));
096        appendValue(buf, depth, "SupportedOperations", supportedOperations);
097    }
098}
