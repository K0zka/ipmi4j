005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011import org.anarres.ipmi.protocol.packet.common.AbstractWireable;
012import org.anarres.ipmi.protocol.packet.common.Code;
013
014/**
015 *
016 * @author shevek
017 */
018// TODO: Refactor with {@link IpmiAlgorithmPayload}?
019public class AsfRsspSessionAuthentication {
020
021    public interface Payload {
022
023        public void toWire(ByteBuffer buffer);
024    }
025
026    /* TODO: Refactor with {@link IpmiAuthenticationAlgorithm}? */
027    public enum AuthenticationAlgorithm implements Payload, Code.Wrapper {
028
029        RAKP_HMAC_SHA1(0x01);
030        public static final short LENGTH = 8;
031        private final byte code;
032
033        private AuthenticationAlgorithm(@Nonnegative int code) {
034            this.code = UnsignedBytes.checkedCast(code);
035        }
036
037        @Override
038        public byte getCode() {
039            return code;
040        }
041
042        @Override
043        public void toWire(ByteBuffer buffer) {
044            buffer.put((byte) 0x01);   // Payload type AuthenticationAlgorithm
045            buffer.put((byte) 0x00);   // Reserved
046            buffer.putShort(LENGTH);
047            buffer.put(code);
048            buffer.put(new byte[3]);    // Reserved
049        }
050    }
051
052    /* TODO: Refactor with {@link IpmiIntegrityAlgorithm}? */
053    public enum IntegrityAlgorithm implements Payload, Code.Wrapper {
054
055        HMAC_SHA1_96(0x01);
056        public static final short LENGTH = 8;
057        private final byte code;
058
059        private IntegrityAlgorithm(int code) {
060            this.code = UnsignedBytes.checkedCast(code);
061        }
062
063        @Override
064        public byte getCode() {
065            return code;
066        }
067
068        @Override
069        public void toWire(ByteBuffer buffer) {
070            buffer.put((byte) 0x02);   // Payload type IntegrityAlgorithm
071            buffer.put((byte) 0x00);   // Reserved
072            buffer.putShort(LENGTH);
073            buffer.put(code);
074            buffer.put(new byte[3]);    // Reserved
075        }
076    }
077
078    public enum EndOfList implements Payload {
079
080        INSTANCE;
081        public static final short LENGTH = 4;
082
083        @Override
084        public void toWire(ByteBuffer buffer) {
085            buffer.put((byte) 0x00);   // Payload type EndOfList
086            buffer.put((byte) 0x00);   // Reserved
087            buffer.putShort(LENGTH);
088        }
089    };
090
091    @Nonnull
092    public static Payload fromWire(@Nonnull ByteBuffer buffer) {
093        if (!buffer.hasRemaining())
094            return EndOfList.INSTANCE;
095
096        byte type = buffer.get();
097        AbstractWireable.assertWireByte(buffer, (byte) 0, "reserved byte");
098        short length = buffer.getShort();
099        byte[] data = new byte[length - 4];
100        buffer.get(data);
101
102        switch (buffer.get()) {
103            case 0:
104                return EndOfList.INSTANCE;
105            case 1:
106                return Code.fromByte(AuthenticationAlgorithm.class, data[0]);
107            case 2:
108                return Code.fromByte(IntegrityAlgorithm.class, data[0]);
109            default:
110                throw new IllegalArgumentException("Unknown algorithm type 0x" + UnsignedBytes.toString(type, 16));
111        }
112    }
113}
