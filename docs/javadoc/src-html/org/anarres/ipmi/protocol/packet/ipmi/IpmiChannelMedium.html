061    private final byte code;
062    private final String description;
063
064    private IpmiChannelMedium(int code, String description) {
065        this.code = UnsignedBytes.checkedCast(code);
066        this.description = description;
067    }
068
069    @Override
070    public byte getCode() {
071        return code;
072    }
073
074    @Override
075    public String getDescription() {
076        return description;
077    }
078
079    @Override
080    public String toString() {
081        return Code.Utils.toString(this);
082    }
