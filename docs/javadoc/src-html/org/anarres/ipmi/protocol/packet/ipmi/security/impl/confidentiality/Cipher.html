005package org.anarres.ipmi.protocol.packet.ipmi.security.impl.confidentiality;
006
007import java.nio.ByteBuffer;
008import java.security.InvalidAlgorithmParameterException;
009import java.security.InvalidKeyException;
010import java.security.Key;
011import java.security.spec.AlgorithmParameterSpec;
012import javax.annotation.Nonnegative;
013import javax.annotation.Nonnull;
014import javax.crypto.ShortBufferException;
015import org.anarres.ipmi.protocol.packet.ipmi.security.IpmiConfidentialityAlgorithm;
016
017/**
018 *
019 * @author shevek
020 */
021public interface Cipher {
022
023    public enum Mode {
024
025        ENCRYPT, DECRYPT;
026    }
027
028    @Nonnull
029    public IpmiConfidentialityAlgorithm getName();
030
031    @Nonnegative
032    public int getIVSize();
033
034    @Nonnegative
035    public int getBlockSize();
036
037    /**
038     * @see javax.crypto.Cipher#init(int, Key, AlgorithmParameterSpec)
039     */
040    public void init(@Nonnull Mode mode, @Nonnull byte[] key, @Nonnull byte[] iv)
041            throws InvalidKeyException, InvalidAlgorithmParameterException;
042
043    /**
044     * @see javax.crypto.Cipher#update(ByteBuffer, ByteBuffer)
045     */
046    public int update(@Nonnull ByteBuffer input, @Nonnull ByteBuffer output)
047            throws ShortBufferException;
048}
