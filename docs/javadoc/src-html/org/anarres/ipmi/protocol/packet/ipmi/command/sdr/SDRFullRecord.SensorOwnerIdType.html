028        IPMBSlaveAddress, SystemSoftwareId;

031    public byte sensorNumber;
032    // Record body.
033    public SDREntityId entityId;
034
035    public enum EntityInstanceType {
036
037        Physical, Logical;
038    }
039    public EntityInstanceType entityInstanceType;
040    // Page 525.
041    /** 7 bits only. */
042    public byte entityInstanceNumber;
043
044    public enum SensorInitialization implements Bits.Wrapper {
045
046        Settable(7, true), NotSettable(7, false),
047        InitScanning(6, true),
048        InitEvents(5, true),
049        InitThresholds(4, true),
050        InitHysteresis(3, true),
051        InitSensorType(2, true),
052        EventGenerationDisabled(1, false), EventGenerationEnabled(1, true),
053        SensorScanningDisabled(0, false), SensorScanningEnabled(0, true);
054        private final Bits bits;
055
056        /* pp */ SensorInitialization(@Nonnegative int bit, boolean value) {
057            this.bits = Bits.forBitIndex(0, bit, value);
058        }
059
060        @Override

062            return bits;
063        }
064    }
065    public Set<SensorInitialization> sensorInitialization = EnumSet.noneOf(SensorInitialization.class);
066
067    public enum SensorCapabilities implements Bits.Wrapper {
068
069        IgnoreIfNotPresent(Bits.forBitIndex(0, 7)),
070        SensorRearmManual(Bits.forBitIndex(0, 6, false)), SensorRearmAutomatic(Bits.forBitIndex(0, 6, true)),
071        HysteresisNone(new Bits(0, 0b11 << 4, 0b00 << 4)),
072        HysteresisReadable(new Bits(0, 0b11 << 4, 0b01 << 4)),
073        HysteresisReadableAndSettable(new Bits(0, 0b11 << 4, 0b10 << 4)),
074        HysteresisFixed(new Bits(0, 0b11 << 4, 0b11 << 4)),
075        ThresholdsNone(new Bits(0, 0b11 << 2, 0b00 << 2)),
076        ThresholdsReadable(new Bits(0, 0b11 << 2, 0b01 << 2)),
077        ThresholdsReadableAndSettable(new Bits(0, 0b11 << 2, 0b10 << 2)),
078        ThresholdsFixed(new Bits(0, 0b11 << 2, 0b11 << 2)),
079        EventMessageControlPerThreshold(new Bits(0, 0b11 << 2, 0b00 << 2)),
080        EventMessageControlPerSensor(new Bits(0, 0b11 << 2, 0b01 << 2)),
081        EventMessageControlGlobal(new Bits(0, 0b11 << 2, 0b10 << 2)),
082        EventMessageControlNoEvents(new Bits(0, 0b11 << 2, 0b11 << 2));
083        private final Bits bits;
084
085        /* pp */ SensorCapabilities(Bits bits) {
086            this.bits = bits;
087        }
088
089        @Override
090        public Bits getBits() {
091            return bits;
092        }
093    }
094    // Page 526.

113    }
