129    /* pp */ SDRDeviceType(@Nonnegative int code, @Nonnull String description, @CheckForNull Class<? extends SDRDeviceSubtype> subtype) {
130        this.code = UnsignedBytes.checkedCast(code);
131        this.description = description;
132        this.subtype = subtype;
133    }
134
135    /* pp */ SDRDeviceType(@Nonnegative int code, @Nonnull String description) {
136        this(code, description, null);
137    }
138
139    @Override
140    public byte getCode() {
141        return code;
142    }
143
144    @Override
145    public String getDescription() {
146        return description;
147    }
148
149    @Nonnull
150    public SDRDeviceSubtype getSubtype(byte code) {
151        // SDRDeviceSubtype doesn't capture that it's an enum.
152        if (subtype != null)
153            for (SDRDeviceSubtype value : subtype.getEnumConstants())
154                if (value.getCode() == code)
155                    return value;
156        return new SDRDeviceSubtype.Unknown(code);
157    }
158
159    @Override
160    public String toString() {
161        return "0x" + UnsignedBytes.toString(getCode(), 16) + ": " + getDescription();
162    }
