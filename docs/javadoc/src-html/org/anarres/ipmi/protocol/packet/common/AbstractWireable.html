007import com.google.common.base.Preconditions;
008import com.google.common.primitives.Chars;
009import com.google.common.primitives.Ints;

012import java.nio.ByteBuffer;
013import java.util.UUID;
014import javax.annotation.CheckForNull;
015import javax.annotation.Nonnegative;
016import javax.annotation.Nonnull;

021/**
022 *
023 * @author shevek
024 */
025public abstract class AbstractWireable implements Wireable {
026
027    private static final Logger LOG = LoggerFactory.getLogger(AbstractWireable.class);
028
029    protected abstract void toWireUnchecked(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer);
030
031    @Override
032    public void toWire(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer) {
033        Preconditions.checkNotNull(buffer, "ByteBuffer was null.");
034        int start = buffer.position();
035        toWireUnchecked(context, buffer);
036
037        int expectedLength = getWireLength(context);
038        int actualLength = buffer.position() - start;
039        if (actualLength != expectedLength)
040            throw new IllegalStateException("Object should serialize to " + expectedLength + " bytes, but generated " + actualLength + ": " + this);
041    }
042
043    protected abstract void fromWireUnchecked(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer);
044
045    @Override
046    public void fromWire(@Nonnull IpmiPacketContext context, ByteBuffer buffer) {
047        Preconditions.checkNotNull(buffer, "ByteBuffer was null.");
048        int start = buffer.position();
049        fromWireUnchecked(context, buffer);
050
051        int expectedLength = getWireLength(context);
052        int actualLength = buffer.position() - start;
053        if (actualLength != expectedLength)
054            throw new IllegalStateException("Object should deserialize from " + expectedLength + " bytes, but consumed " + actualLength + ": " + this.getClass() + ": " + this);
055        // LOG.debug("Decoded " + this.getClass().getSimpleName() + " from " + actualLength + " bytes.");
056    }
057
058    /** Reads an array of bytes from the wire and returns them. */
059    @Nonnull
060    public static byte[] readBytes(@Nonnull ByteBuffer buffer, @Nonnegative int length) {
061        byte[] data = new byte[length];
062        buffer.get(data);
063        return data;
064    }
065
066    // TODO: Take a 'description' message too.
067    /** Reads a 4-byte signed int from the wire, and asserts it equal to the given expected value. */
068    public static void assertWireInt(@Nonnull ByteBuffer buffer, int expectValue, String description) {
069        int actualValue = buffer.getInt();
070        if (actualValue != expectValue)
071            throw new IllegalArgumentException("In " + description + ": "
072                    + "Expected 0x" + Integer.toHexString(expectValue)
073                    + " but got 0x" + Integer.toHexString(actualValue));
074    }
075
076    /** Reads a 2-byte unsigned char from the wire, and asserts it equal to the given expected value. */
077    public static void assertWireChar(@Nonnull ByteBuffer buffer, char expectValue, String description) {
078        short actualValue = buffer.get();
079        if (actualValue != expectValue)
080            throw new IllegalArgumentException("In " + description + ": "
081                    + "Expected 0x" + Integer.toHexString(expectValue)
082                    + " but got 0x" + Integer.toHexString(actualValue));
083    }
084
085    /** Reads a byte from the wire, and asserts it equal to the given expected value. */
086    public static void assertWireByte(@Nonnull ByteBuffer buffer, byte expectValue, @Nonnull String description) {
087        byte actualValue = buffer.get();
088        if (actualValue != expectValue)
089            throw new IllegalArgumentException("In " + description + ": "
090                    + "Expected 0x" + UnsignedBytes.toString(expectValue, 16)
091                    + " but got 0x" + UnsignedBytes.toString(actualValue, 16));
092    }
093
094    /** Reads a number of bytes from the wire, and asserts them equal to the given expected values. */
095    public static void assertWireBytes(@Nonnull ByteBuffer buffer, @Nonnull int... expectValues) {
096        for (int expectValue : expectValues)
097            assertWireByte(buffer, (byte) (expectValue & 0xFF), "data byte at offset " + buffer.position());
098    }
099
100    /** Reads a number of zero bytes from the buffer. */
101    public static void assertWireBytesZero(@Nonnull ByteBuffer buffer, @Nonnegative int count) {
102        for (int i = 0; i < count; i++)
103            assertWireByte(buffer, (byte) 0, "data byte at offset " + buffer.position() + " (reserved byte " + i + ")");
104    }
105
106    protected int assertMask(int value, int nbits) {
107        if (Integer.numberOfLeadingZeros(value) < Integer.SIZE - nbits)
108            throw new IllegalArgumentException("Too many bits in " + Integer.toHexString(value) + "; expected max " + nbits);
109        return value;
110    }
111
112    /** [IPMI2] Section 20.8, table 20-10, page 252. */
113    public static void toWireUUIDLE(@Nonnull ByteBuffer buf, @Nonnull UUID uuid) {
114        toWireLongLE(buf, uuid.getLeastSignificantBits());
115        toWireLongLE(buf, uuid.getMostSignificantBits());
116    }
117
118    /** [IPMI2] Section 20.8, table 20-10, page 252. */
119    @Nonnull
120    public static UUID fromWireUUIDLE(@Nonnull ByteBuffer buf) {
121        long lsb = fromWireLongLE(buf);
122        long msb = fromWireLongLE(buf);
123        return new UUID(msb, lsb);
124    }
125
126    public static long fromWireLongLE(@Nonnull ByteBuffer buffer) {
127        byte b0 = buffer.get();
128        byte b1 = buffer.get();
129        byte b2 = buffer.get();
130        byte b3 = buffer.get();
131        byte b4 = buffer.get();
132        byte b5 = buffer.get();
133        byte b6 = buffer.get();
134        byte b7 = buffer.get();
135        return Longs.fromBytes(b7, b6, b5, b4, b3, b2, b1, b0);
136    }
137
138    public static void toWireLongLE(@Nonnull ByteBuffer buffer, long c) {
139        buffer.put((byte) (c));
140        buffer.put((byte) (c >> 8));
141        buffer.put((byte) (c >> 16));
142        buffer.put((byte) (c >> 24));
143        buffer.put((byte) (c >> 32));
144        buffer.put((byte) (c >> 40));
145        buffer.put((byte) (c >> 48));
146        buffer.put((byte) (c >> 56));
147    }
148
149    public static void toWireIntLE(@Nonnull ByteBuffer buffer, int data) {
150        buffer.put((byte) (data));
151        buffer.put((byte) (data >> 8));
152        buffer.put((byte) (data >> 16));
153        buffer.put((byte) (data >> 24));
154    }
155
156    public static int fromWireIntLE(@Nonnull ByteBuffer buffer) {
157        byte b0 = buffer.get();
158        byte b1 = buffer.get();
159        byte b2 = buffer.get();
160        byte b3 = buffer.get();
161        return Ints.fromBytes(b3, b2, b1, b0);
162    }
163
164    public static void toWireCharLE(@Nonnull ByteBuffer buffer, char data) {
165        buffer.put((byte) (data));
166        buffer.put((byte) (data >> 8));
167    }
168
169    public static char fromWireCharLE(@Nonnull ByteBuffer buffer) {
170        byte b0 = buffer.get();
171        byte b1 = buffer.get();
172        return Chars.fromBytes(b1, b0);
173    }
174
175    @Nonnull
176    protected static StringBuilder indent(@Nonnull StringBuilder buf, int depth) {
177        for (int i = 0; i < depth; i++)
178            buf.append("  ");
179        return buf;
180    }
181
182    @Nonnull
183    public static String toHexString(@CheckForNull byte... data) {
184        if (data == null)
185            return "<null>";
186        StringBuilder buf = new StringBuilder();
187        buf.append("(").append(data.length).append(" bytes) ");
188        for (byte b : data) {
189            String s = UnsignedBytes.toString(b, 16);
190            if (s.length() < 2)
191                buf.append('0');
192            buf.append(s).append(' ');
193        }
194        buf.setLength(buf.length() - 1);
195        return buf.toString();
196    }
197
198    @Nonnull
199    public static String toBinaryString(@CheckForNull byte... data) {
200        if (data == null)
201            return "<null>";
202        StringBuilder buf = new StringBuilder();
203        buf.append("(").append(data.length).append(" bytes) ");
204        for (byte b : data) {
205            String s = Integer.toBinaryString(b & 0xFF);
206            for (int i = s.length(); i < 8; i++)
207                buf.append('0');
208            buf.append(s).append(' ');
209        }
210        buf.setLength(buf.length() - 1);
211        return buf.toString();
212    }
213
214    @Nonnull
215    protected static StringBuilder appendHeader(@Nonnull StringBuilder buf, int depth, String name) {
216        indent(buf, depth);
217        buf.append(name).append(":").append('\n');
218        return buf;
219    }
220
221    @Nonnull
222    protected static StringBuilder appendValue(@Nonnull StringBuilder buf, int depth, String name, Object value) {
223        indent(buf, depth);
224        buf.append(name).append(": ").append(value).append('\n');
225        return buf;
226    }
227
228    // protected static StringBuilder appendCode(@Nonnull StringBuilder buf, int depth, String name, Class<?> type, byte code) { }
229    @Nonnull
230    protected static StringBuilder appendChild(@Nonnull StringBuilder buf, int depth, @Nonnull String name, @CheckForNull Wireable value) {
231        if (value == null) {
232            appendHeader(buf, depth, name);
233            indent(buf, depth).append("null\n");
234        } else {
235            appendValue(buf, depth, name, value.getClass().getSimpleName());
236            value.toStringBuilder(buf, depth + 1);
237        }
238        return buf;
239    }
240
241    @Override
242    public void toStringBuilder(StringBuilder buf, int depth) {
243        indent(buf, depth).append(super.toString()).append('\n');
244    }
245
246    @Override
247    public String toString() {
248        StringBuilder buf = new StringBuilder();
249        toStringBuilder(buf, 0);
250        return buf.toString();
251    }
