132    @Override
133    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
134        int chk1Start = buffer.position();
135        buffer.put(getTargetAddress());
136        byte networkFunctionByte = getCommandName().getNetworkFunction().getCode();
137        if (this instanceof IpmiResponse)
138            networkFunctionByte |= 1;
139        buffer.put((byte) (networkFunctionByte << 2 | getTargetLun().getValue()));
140        toWireChecksum(buffer, chk1Start);
141        int chk2Start = buffer.position();
142        buffer.put(getSourceAddress());
143        int seq = getSequenceNumber() & SEQUENCE_NUMBER_MASK;
144        buffer.put((byte) (seq << 2 | getSourceLun().getValue()));
145        buffer.put(getCommandName().getCode());
146        toWireData(buffer);
147        toWireChecksum(buffer, chk2Start);
148    }

150    protected abstract void toWireData(@Nonnull ByteBuffer buffer);

152    /**
153     * {@inheritDoc}
154     *
155     * This implementation must be called with a buffer of exactly decodeable
156     * length, as some packet types require consumption of "optional" or
157     * "remaining" data without an auxiliary embedded length field.
158     *
159     * @see AbstractIpmiSessionWrapper#newPayload(ByteBuffer, IpmiPayloadType)
160     */
161    @Override
162    protected void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
163        int chk1Start = buffer.position();
164        int tmp;
165        targetAddress = buffer.get();
166        tmp = UnsignedBytes.toInt(buffer.get());
167        IpmiNetworkFunction networkFunction = Code.fromInt(IpmiNetworkFunction.class, (tmp >>> 2) & ~1);
168        targetLun = Code.fromInt(IpmiLun.class, tmp & IpmiLun.MASK);
169        fromWireChecksum(buffer, chk1Start, "IPMI header checksum");
170        int chk2Start = buffer.position();
171        sourceAddress = buffer.get();
172        tmp = UnsignedBytes.toInt(buffer.get());
173        sequenceNumber = (byte) ((tmp >>> 2) & SEQUENCE_NUMBER_MASK);
174        sourceLun = Code.fromInt(IpmiLun.class, tmp & IpmiLun.MASK);
175        IpmiCommandName commandName = IpmiCommandName.fromByte(networkFunction, buffer.get());
176        buffer.limit(buffer.limit() - 1);   // Represent an accurate data length to the packet data decoder.
177        fromWireData(buffer);
178        buffer.limit(buffer.limit() + 1);   // And let us get the checksum out.
179        fromWireChecksum(buffer, chk2Start, "IPMI data checksum");
180    }

184    /** [IPMI2] Section 13.8, page 137. */
185    private static byte toChecksum(@Nonnull ByteBuffer buffer, @Nonnegative int start) {
186        int csum = 0;
187        for (int i = start; i < buffer.position(); i++)
188            csum += buffer.get(i);
189        return (byte) -csum;
190    }
191
192    public static void toWireChecksum(@Nonnull ByteBuffer buffer, @Nonnegative int start) {
193        buffer.put(toChecksum(buffer, start));
194    }
195
196    public void fromWireChecksum(@Nonnull ByteBuffer buffer, @Nonnegative int start, @Nonnull String description) {
197        byte expect = toChecksum(buffer, start);
198        byte actual = buffer.get();
199        if (expect != actual)
200            throw new IllegalArgumentException("Checksum failure: " + description
201                    + ": expected=" + UnsignedBytes.toString(expect, 16)
202                    + " actual=" + UnsignedBytes.toString(actual, 16) + "; command=" + getClass().getSimpleName() + "; data=" + this);
203    }

221    @Nonnull
222    public static String toStringOemIana(int code) {
223        IanaEnterpriseNumber value = null;
224        for (IanaEnterpriseNumber n : IanaEnterpriseNumber.values()) {
225            if (code == n.getNumber()) {
226                value = n;
227                break;
228            }
229        }
230        return "0x" + Integer.toHexString(code) + " (" + code + "): " + value;
231    }

233    /**
234     * Encodes one byte (two digits) of BCD.
235     * The argument cannot be negative as the domain is 0-99.
236     */
237    public static byte toWireBcdLE(@Nonnegative byte data) {
238        Preconditions.checkPositionIndex(data, 99, "BCD encoder input out of range");
239        int b0 = data % 10; // Low digit
240        data /= 10;
241        int b1 = data % 10; // High digit
242        return (byte) ((b0 & 0xF) << 4 | (b1 & 0xF));
243    }
244
245    @Nonnegative
246    public static void toWireBcdLE(@Nonnull ByteBuffer buffer, @Nonnegative byte data) {
247        buffer.put(toWireBcdLE(data));
248    }

250    /**
251     * Decodes one byte (two digits) of BCD.
252     * The return value cannot be negative as the range is 0-99.
253     */
254    @Nonnegative
255    public static byte fromWireBcdLE(byte data) {
256        int b0 = (data >> 4) & 0xF; // Low digit
257        int b1 = data & 0xF; // High digit
258        Preconditions.checkPositionIndex(b0, 9, "BCD decoder low digit out of range");
259        Preconditions.checkPositionIndex(b1, 9, "BCD decoder high digit out of range");
260        return (byte) (b1 * 10 + b0);
261    }
262
263    @Nonnegative
264    public static byte fromWireBcdLE(@Nonnull ByteBuffer buffer) {
265        return fromWireBcdLE(buffer.get());
266    }
