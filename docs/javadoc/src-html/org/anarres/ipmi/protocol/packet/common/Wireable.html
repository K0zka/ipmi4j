005package org.anarres.ipmi.protocol.packet.common;
006
007import java.nio.ByteBuffer;
008import javax.annotation.Nonnegative;
009import javax.annotation.Nonnull;
010import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
011
012/**
013 * An object which may be written to and read from a {@link ByteBuffer}.
014 *
015 * @author shevek
016 */
017public interface Wireable {
018
019    /**
020     * Returns the wire length of this packet.
021     *
022     * This must be exact, and not an overestimate.
023     * If {@link #toWire(IpmiPacketContext, java.nio.ByteBuffer)} writes fewer bytes than the
024     * computed length, the base class will assume error in the serializer
025     * and throw an exception.
026     * 
027     * It is generally not valid to call getWireLength(IpmiPacketContext) on a Wireable which
028     * does not contain complete data, as the length of data-dependent fields
029     * cannot be computed.
030         *
031         * @param context The IpmiPacketContext for this operation.
032     */
033    @Nonnegative
034    public int getWireLength(@Nonnull IpmiPacketContext context);
035
036    /**
037     * Writes bytes from this object onto the wire, serializing it.
038         *
039         * @param context The IpmiPacketContext for this operation.
040         * @param buffer The ByteBuffer into which this Wireable will be serialized.
041     */
042    public void toWire(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer);
043
044    /**
045     * Reads bytes from the wire into this object, deserializing it.
046         *
047         * @param context The IpmiPacketContext for this operation.
048         * @param buffer The ByteBuffer from which this Wireable will be deserialized.
049     */
050    public void fromWire(@Nonnull IpmiPacketContext context, @Nonnull ByteBuffer buffer);
051
052    public void toStringBuilder(@Nonnull StringBuilder buf, @Nonnegative int depth);
053}
