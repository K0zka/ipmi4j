005package org.anarres.ipmi.protocol.packet.ipmi.command.messaging;
006
007import java.nio.ByteBuffer;
008import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
009import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
010import org.anarres.ipmi.protocol.packet.common.Code;
011import org.anarres.ipmi.protocol.packet.ipmi.IpmiChannelNumber;
012import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
013import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiRequest;
014import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayloadType;
015
016/**
017 * [IPMI2] Section 22.15, table 22-17, page 290.
018 *
019 * @author shevek
020 */
021public class GetChannelCipherSuitesRequest extends AbstractIpmiRequest {
022
023    public enum ListType {
024
025        ByCipherSuite,
026        Supported
027    }
028    public IpmiChannelNumber channelNumber;
029    public IpmiPayloadType payloadType;
030    public ListType listType = ListType.Supported;
031    public int listIndex;
032
033    @Override
034    public IpmiCommandName getCommandName() {
035        return IpmiCommandName.GetChannelCipherSuites;
036    }
037
038    @Override
039    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
040        handler.handleGetChannelCipherSuitesRequest(context, this);
041    }
042
043    @Override
044    protected int getDataWireLength() {
045        return 3;
046    }
047
048    @Override
049    protected void toWireData(ByteBuffer buffer) {
050        buffer.put(channelNumber.getCode());
051        buffer.put(payloadType.getCode());
052        byte tmp = (byte) (listIndex & 0x3F);
053        tmp = setBit(tmp, 7, listType == ListType.ByCipherSuite);
054        buffer.put(tmp);
055    }
056
057    @Override
058    protected void fromWireData(ByteBuffer buffer) {
059        channelNumber = Code.fromBuffer(IpmiChannelNumber.class, buffer);
060        payloadType = Code.fromBuffer(IpmiPayloadType.class, buffer);
061        byte tmp = buffer.get();
062        listIndex = tmp & 0x3F;
063        listType = getBit(tmp, 7) ? ListType.ByCipherSuite : ListType.Supported;
064    }
065
066    @Override
067    public void toStringBuilder(StringBuilder buf, int depth) {
068        super.toStringBuilder(buf, depth);
069        appendValue(buf, depth, "ChannelNumber", channelNumber);
070        appendValue(buf, depth, "PayloadType", payloadType);
071        appendValue(buf, depth, "ListIndex", listIndex);
072        appendValue(buf, depth, "ListType", listType);
073    }
074}
