005package org.anarres.ipmi.protocol.packet.ipmi.security.impl.integrity;
006
007import javax.annotation.Nonnegative;
008import javax.annotation.Nonnull;
009
010/**
011 * [IPMI2] Section 13.6, table 13-8, page 134.
012 * 
013 * Added as needed to cause the number of bytes in the data range covered by
014 * the AuthCode (Integrity Data) field to be a multiple of 4 bytes (DWORD).
015 * If present, each Integrity Pad byte is set to FFh.
016 *
017 * @author shevek
018 */
019public class IntegrityPad {
020
021    private static final byte[] PAD0 = new byte[]{};
022    private static final byte[] PAD1 = new byte[]{(byte) 0xFF};
023    private static final byte[] PAD2 = new byte[]{(byte) 0xFF, (byte) 0xFF};
024    private static final byte[] PAD3 = new byte[]{(byte) 0xFF, (byte) 0xFF, (byte) 0xFF};
025
026    @Nonnull
027    public static byte[] PAD(@Nonnegative int length) {
028        switch (length & 0x03) {
029            case 0:
030                return PAD0;
031            case 1:
032                return PAD3;
033            case 2:
034                return PAD2;
035            case 3:
036                return PAD1;
037            default:
038                throw new IllegalStateException("Illegal length " + length);
039        }
040    }
041}
