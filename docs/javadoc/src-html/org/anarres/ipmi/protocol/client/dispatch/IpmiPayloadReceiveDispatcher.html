005package org.anarres.ipmi.protocol.client.dispatch;
006
007import com.google.common.annotations.VisibleForTesting;
008import com.google.common.base.Preconditions;
009import com.google.common.cache.Cache;
010import com.google.common.cache.CacheBuilder;
011import com.google.common.cache.RemovalListener;
012import com.google.common.cache.RemovalNotification;
013import java.util.concurrent.TimeUnit;
014import javax.annotation.Nonnull;
015import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
016import org.anarres.ipmi.protocol.client.session.IpmiSession;
017import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiPayloadHandler;
018import org.anarres.ipmi.protocol.client.visitor.IpmiClientRmcpMessageHandler;
019import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
020import org.anarres.ipmi.protocol.packet.ipmi.IpmiSessionWrapper;
021import org.anarres.ipmi.protocol.packet.ipmi.command.IpmiCommand;
022import org.anarres.ipmi.protocol.packet.ipmi.command.IpmiResponse;
023import org.anarres.ipmi.protocol.packet.ipmi.payload.AbstractTaggedIpmiPayload;
024import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayload;
025import org.anarres.ipmi.protocol.packet.rmcp.Packet;
026import org.anarres.ipmi.protocol.packet.rmcp.RmcpData;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 *
032 * @author shevek
033 */
034public class IpmiPayloadReceiveDispatcher implements IpmiReceiverRepository {
035
036    private static final Logger LOG = LoggerFactory.getLogger(IpmiPayloadReceiveDispatcher.class);
037
038    private final IpmiPacketContext sessionManager;
039    private final RemovalListener<IpmiReceiverKey, IpmiReceiver> removalListener = new RemovalListener<IpmiReceiverKey, IpmiReceiver>() {
040
041        @Override
042        public void onRemoval(RemovalNotification<IpmiReceiverKey, IpmiReceiver> notification) {
043            IpmiReceiverKey key = notification.getKey();
044            IpmiReceiver receiver = notification.getValue();
045            if (key != null && receiver != null && notification.wasEvicted())
046                receiver.timeout(key);
047        }
048    };
049    /** [IPMI2] Section 6.12.15, page 60: Session inactivity timeout is 60 seconds +/-3 seconds. */
050    private final Cache<IpmiReceiverKey, IpmiReceiver> ipmiReceivers = CacheBuilder.newBuilder()
051            // .maximumSize(64)
052            .expireAfterWrite(180, TimeUnit.SECONDS)
053            .removalListener(removalListener)
054            .recordStats()
055            .build();
056    /* First, for RMCP-land. */
057    private final IpmiClientRmcpMessageHandler rmcpHandler = new IpmiClientRmcpMessageHandler.Adapter() {
058
059        @Override
060        public void handleDefault(IpmiHandlerContext context, RmcpData message) {
061            handleDiscard(context, message);
062        }
063
064        @Override
065        public void handleIpmiRmcpData(IpmiHandlerContext context, IpmiSessionWrapper message) {
066            // Pass directly to IpmiPayload to avoid type ambiguity on 'this'.
067            int sessionId = message.getIpmiSessionId();
068            IpmiSession session = (sessionId == 0) ? null : sessionManager.getIpmiSession(sessionId);
069            message.getIpmiPayload().apply(ipmiPayloadHandler, context, session);
070        }
071    };
072    /* And now for IPMI-land. */
073    private IpmiClientIpmiPayloadHandler ipmiPayloadHandler = new IpmiClientIpmiPayloadHandler.TaggedAdapter() {
074
075        @Override
076        protected void handleDefault(IpmiHandlerContext context, IpmiSession session, IpmiPayload payload) {
077            handleDiscard(context, payload);
078        }
079
080        @Override
081        protected void handleTagged(@Nonnull IpmiHandlerContext context, @Nonnull IpmiSession session, @Nonnull AbstractTaggedIpmiPayload payload) {
082            IpmiReceiver receiver = (IpmiReceiver) getReceiver(context, payload.getClass(), payload.getMessageTag());
083            if (receiver != null) {
084                receiver.receive(context, session, payload);
085                return;
086            }
087            handleDiscard(context, payload);
088        }
089
090        /** Section 6.12.8, page 58: Requests and responses are matched on the IPMI Seq field. */
091        @Override
092        public void handleCommand(IpmiHandlerContext context, IpmiSession session, IpmiCommand message) {
093            if (message instanceof IpmiResponse) {
094                IpmiResponse response = (IpmiResponse) message;
095                IpmiReceiver receiver = getReceiver(context, response.getClass(), message.getSequenceNumber());
096                if (receiver != null) {
097                    receiver.receive(context, session, response);
098                    return;
099                }
100            }
101            handleDiscard(context, message);
102        }
103    };
104
105    public IpmiPayloadReceiveDispatcher(@Nonnull IpmiPacketContext sessionManager) {
106        this.sessionManager = sessionManager;
107    }
108
109    @Override
110    public IpmiReceiver getReceiver(@Nonnull IpmiHandlerContext context, Class<? extends IpmiPayload> payloadType, byte messageId) {
111        IpmiReceiverKey key = new IpmiReceiverKey(context.getSystemAddress(), payloadType, messageId);
112        IpmiReceiver receiver = ipmiReceivers.asMap().remove(key);
113        if (LOG.isDebugEnabled())
114            LOG.debug("Get: " + key + " -> " + receiver);
115        return receiver;
116    }
117
118    @Override
119    public void setReceiver(@Nonnull IpmiReceiverKey key, @Nonnull IpmiReceiver receiver) {
120        if (LOG.isDebugEnabled())
121            LOG.debug("Set: " + key + " -> " + receiver);
122        ipmiReceivers.put(key, receiver);
123    }
124
125    @VisibleForTesting
126    protected void handleDiscard(@Nonnull IpmiHandlerContext context, @Nonnull Object message) {
127        Preconditions.checkNotNull(message, "Message was null.");
128        LOG.warn("Discarded:\n" + message);
129    }
130
131    public void dispatch(@Nonnull IpmiHandlerContext context, @Nonnull Packet packet) {
132        packet.apply(rmcpHandler, context);
133    }
134}
