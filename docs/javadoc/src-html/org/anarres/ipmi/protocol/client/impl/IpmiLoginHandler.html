005package org.anarres.ipmi.protocol.client.impl;
006
007import javax.annotation.Nonnull;
008import org.anarres.ipmi.protocol.client.dispatch.IpmiReceiver;
009import org.anarres.ipmi.protocol.client.dispatch.IpmiReceiverKey;
010import org.anarres.ipmi.protocol.client.session.IpmiSession;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiPayloadHandler;
013import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
014import org.anarres.ipmi.protocol.packet.ipmi.command.IpmiCommand;
015import org.anarres.ipmi.protocol.packet.ipmi.command.messaging.CloseSessionRequest;
016import org.anarres.ipmi.protocol.packet.ipmi.command.messaging.CloseSessionResponse;
017import org.anarres.ipmi.protocol.packet.ipmi.command.messaging.GetChannelAuthenticationCapabilitiesRequest;
018import org.anarres.ipmi.protocol.packet.ipmi.command.messaging.GetChannelAuthenticationCapabilitiesResponse;
019import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiOpenSessionRequest;
020import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiOpenSessionResponse;
021import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayload;
022import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiRAKPMessage1;
023import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiRAKPMessage2;
024import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiRAKPMessage3;
025import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiRAKPMessage4;
026import org.anarres.ipmi.protocol.packet.ipmi.payload.RequestedMaximumPrivilegeLevel;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 *
032 * @author shevek
033 */
034public class IpmiLoginHandler {
035
036    private static final Logger LOG = LoggerFactory.getLogger(IpmiLoginHandler.class);
037    private final IpmiClientIpmiCommandHandler commandHandler = new IpmiClientIpmiCommandHandler.Adapter() {
038        @Override
039        public void handleDefault(IpmiHandlerContext context, IpmiCommand command) {
040            LOG.info("Ignored " + command);
041        }
042
043        // 2
044        @Override
045        public void handleGetChannelAuthenticationCapabilitiesResponse(IpmiHandlerContext context, GetChannelAuthenticationCapabilitiesResponse response) {
046            IpmiSession session = getSession(0, 0);
047            context.send(session, new CloseSessionRequest(session),
048                    CloseSessionResponse.class, receiver);
049        }
050
051        // 3
052        @Override
053        public void handleCloseSessionResponse(IpmiHandlerContext context, CloseSessionResponse response) {
054            IpmiSession session = getSession(0, 0);
055            context.send(null, new IpmiOpenSessionRequest(session, RequestedMaximumPrivilegeLevel.ADMINISTRATOR),
056                    IpmiOpenSessionResponse.class, receiver);
057        }
058    };
059    private final IpmiClientIpmiPayloadHandler payloadHandler = new IpmiClientIpmiPayloadHandler.Adapter() {
060
061        @Override
062        protected void handleDefault(IpmiHandlerContext context, IpmiSession session, IpmiPayload payload) {
063            LOG.info("Ignored " + payload);
064        }
065
066        // 4
067        @Override
068        public void handleOpenSessionResponse(IpmiHandlerContext context, IpmiSession _session, IpmiOpenSessionResponse response) {
069            IpmiSession session = getSession(response.consoleSessionId, 0);
070            session.setSystemSessionId(response.systemSessionId);
071            context.send(session, new IpmiRAKPMessage1(session), IpmiRAKPMessage2.class, receiver);
072        }
073
074        // 5
075        @Override
076        public void handleRAKPMessage2(IpmiHandlerContext context, IpmiSession _session, IpmiRAKPMessage2 message) {
077            IpmiSession session = getSession(message.consoleSessionId, 0);
078            context.send(session, new IpmiRAKPMessage3(), IpmiRAKPMessage4.class, receiver);
079        }
080
081        // 6
082        @Override
083        public void handleRAKPMessage4(IpmiHandlerContext context, IpmiSession _session, IpmiRAKPMessage4 message) {
084            IpmiSession session = getSession(message.consoleSessionId, 0);
085            // Now your session is open.
086        }
087
088        @Override
089        public void handleCommand(IpmiHandlerContext context, IpmiSession session, IpmiCommand message) {
090            message.apply(commandHandler, context);
091        }
092    };
093    private final IpmiReceiver receiver = new IpmiReceiver() {
094
095        @Override
096        public void receive(IpmiHandlerContext context, IpmiSession session, IpmiPayload response) {
097            LOG.warn("Receive: " + response);
098            response.apply(payloadHandler, context, session);
099        }
100
101        @Override
102        public void timeout(IpmiReceiverKey key) {
103            LOG.warn("Timeout: " + key);
104        }
105    };
106    private final IpmiSession session;
107
108    public IpmiLoginHandler(@Nonnull IpmiSession session) {
109        this.session = session;
110    }
111
112    @Nonnull
113    private IpmiSession getSession(int consoleSessionId, int systemSessionId) {
114        if (consoleSessionId != 0)
115            if (consoleSessionId != session.getConsoleSessionId())
116                throw new IllegalArgumentException("Bad consoleSessionId " + Integer.toHexString(consoleSessionId) + "; expected " + Integer.toHexString(session.getConsoleSessionId()));
117        if (systemSessionId != 0)
118            if (systemSessionId != session.getSystemSessionId())
119                throw new IllegalArgumentException("Bad systemSessionId " + Integer.toHexString(systemSessionId) + "; expected " + Integer.toHexString(session.getSystemSessionId()));
120        return session;
121    }
122
123    // 1
124    public void open(@Nonnull IpmiHandlerContext context) {
125        context.send(null, new GetChannelAuthenticationCapabilitiesRequest(),
126                GetChannelAuthenticationCapabilitiesResponse.class, receiver);
127    }
128}
