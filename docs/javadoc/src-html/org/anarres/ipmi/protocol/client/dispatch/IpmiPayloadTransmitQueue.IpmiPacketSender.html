006package org.anarres.ipmi.protocol.client.dispatch;
007
008import com.google.common.cache.CacheBuilder;
009import com.google.common.cache.CacheLoader;
010import com.google.common.cache.LoadingCache;
011import java.net.SocketAddress;
012import java.util.concurrent.LinkedBlockingQueue;
013import java.util.concurrent.TimeUnit;
014import javax.annotation.CheckForNull;
015import javax.annotation.Nonnull;
016import org.anarres.ipmi.protocol.client.session.IpmiSession;
017import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiPayloadHandler;
018import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
019import org.anarres.ipmi.protocol.packet.ipmi.Ipmi15SessionWrapper;
020import org.anarres.ipmi.protocol.packet.ipmi.IpmiSessionWrapper;
021import org.anarres.ipmi.protocol.packet.ipmi.command.IpmiCommand;
022import org.anarres.ipmi.protocol.packet.ipmi.payload.AbstractTaggedIpmiPayload;
023import org.anarres.ipmi.protocol.packet.ipmi.payload.IpmiPayload;
024import org.anarres.ipmi.protocol.packet.rmcp.Packet;
025import org.anarres.ipmi.protocol.packet.rmcp.RmcpPacket;
026
027/**
028 *
029 * @author shevek
030 */
031public class IpmiPayloadTransmitQueue implements IpmiHandlerContext.IpmiPacketQueue {
032
033    public static interface IpmiPacketSender {
034
035        public void send(@Nonnull Packet packet);
036    }
037
038    // -> Queue<QueueItem>
039    private static class Queue extends LinkedBlockingQueue<IpmiPayload> {
040
041        // private final BitSet messageTags = new BitSet(256);
042        /** @see AbstractTaggedIpmiPayload#getMessageTag() */
043        private int nextMessageTag;
044        // private final BitSet sequenceNumbers = new BitSet(IpmiCommand.SEQUENCE_NUMBER_MASK + 1);
045        /** @see IpmiCommand#getSequenceNumber() */
046        private int nextSequenceNumber;
047        private int outstandingRequests;
048    }
049
050    private static class QueueItem {
051
052        private final IpmiSession session;
053        private final IpmiPayload payload;
054        private final IpmiReceiverKey key;
055        private final IpmiReceiver receiver;
056
057        public QueueItem(@CheckForNull IpmiSession session, @Nonnull IpmiPayload payload,
058                @Nonnull IpmiReceiverKey key, @Nonnull IpmiReceiver receiver) {
059            this.session = session;
060            this.payload = payload;
061            this.key = key;
062            this.receiver = receiver;
063        }
064    }
065
066    private static class QueueFactory extends CacheLoader<SocketAddress, Queue> {
067
068        @Override
069        public Queue load(SocketAddress key) throws Exception {
070            return new Queue();
071        }
072    };
073    private final LoadingCache<SocketAddress, Queue> ipmiQueues = CacheBuilder.newBuilder()
074            // .weakKeys() // Discard queues for closed connections?
075            .expireAfterAccess(1, TimeUnit.HOURS)
076            .recordStats()
077            .build(new QueueFactory());
078    private final IpmiClientIpmiPayloadHandler ipmiPayloadSequencer = new IpmiClientIpmiPayloadHandler.TaggedAdapter() {
079
080        @Override
081        protected void handleDefault(IpmiHandlerContext context, IpmiSession session, IpmiPayload payload) {
082            throw new UnsupportedOperationException();
083        }
084
085        @Override
086        protected void handleTagged(IpmiHandlerContext context, IpmiSession session, AbstractTaggedIpmiPayload message) {
087            Queue queue = getState(context);
088            synchronized (queue) {
089                message.setMessageTag((byte) queue.nextMessageTag++);
090                queue.add(message);
091            }
092        }
093
094        @Override
095        public void handleCommand(IpmiHandlerContext context, IpmiSession session, IpmiCommand message) {
096            Queue queue = getState(context);
097            synchronized (queue) {
098                byte sequenceNumber = (byte) queue.nextSequenceNumber++;
099                message.setSequenceNumber(sequenceNumber);
100                // receiver.setReceiver(context, null, sequenceNumber, null);
101                queue.add(message);
102            }
103        }
104    };
105    private final IpmiReceiverRepository receiver;
106    private final IpmiPacketSender sender;
107
108    public IpmiPayloadTransmitQueue(@Nonnull IpmiReceiverRepository receiver, @Nonnull IpmiPacketSender sender) {
109        this.receiver = receiver;
110        this.sender = sender;
111    }
112
113    @Nonnull
114    private Queue getState(@Nonnull IpmiHandlerContext context) {
115        return ipmiQueues.getUnchecked(context.getSystemAddress());
116    }
117
118    @Override
119    public void queue(IpmiHandlerContext context, IpmiSession session, IpmiPayload message,
120            Class<? extends IpmiPayload> responseType, IpmiReceiver receiver) {
121        message.apply(ipmiPayloadSequencer, context, session);
122    }
123
124    private void doSend(@Nonnull SocketAddress systemAddress, @Nonnull QueueItem item) {
125        @CheckForNull
126        IpmiSession session = item.session;
127        @Nonnull
128        IpmiPayload payload = item.payload;
129
130        IpmiSessionWrapper wrapper = new Ipmi15SessionWrapper();
131        wrapper.setIpmiPayload(payload);
132        if (session != null) {
133            if (wrapper.isEncrypted())
134                wrapper.setIpmiSessionSequenceNumber(session.nextEncryptedSequenceNumber());
135            else
136                wrapper.setIpmiSessionSequenceNumber(session.nextUnencryptedSequenceNumber());
137        } else {
138            wrapper.setIpmiSessionSequenceNumber(0);
139        }
140
141        RmcpPacket packet = new RmcpPacket();
142        packet.withRemoteAddress(systemAddress);
143        packet.withData(wrapper);
144
145        receiver.setReceiver(item.key, item.receiver);
146        sender.send(packet);
147    }
148}
